202void ClassDiagramWindow::addClassNode()
203{
204 // Numbering and positioning unique names
205 static unsigned int classNumber = 1;
206 static const unsigned int nodeSpace = 50;
207
208 // Create new class in class diagram
209 Class *newClass = new Class{"New class " + std::to_string(classNumber++), std::make_tuple(0, 0)};
210
211 classDiagram.addClass(newClass);
212
213 // Create new GUI class node
214 auto *newClassNode = new ClassNode(newClass, &storedClasses, sceneUpdateObservable);
215 newClassNode->setPos((newClassNode->boundingRect().width() + nodeSpace) * (classNumber % 5),
216 (newClassNode->boundingRect().height() + nodeSpace) * (classNumber / 5));
217
218 connect(newClassNode->emitor, &ClassNodeEmmitor::nodePressed, this, &ClassDiagramWindow::nodePressed);
219 classDiagramScene->addItem(newClassNode);
220 storedClasses.insert({newClass->getName(), newClassNode});
221
222 sceneUpdateObservable->sceneChanged();
223}

247void ClassDiagramWindow::removeClassNode(ClassNode *classNode)
248{
249 QSet<Line *> connections = classNode->getConnections();
250 for(Line *connection: connections) {
251 // Delete corresponding relationship from class diagram and memory
252 Relationship *relationship = storedRelationships.find(connection)->second;
253
254 classDiagram.removeRelationship(relationship);
255 delete relationship;
256
257 // Delete from stored relationships
258 storedRelationships.erase(connection);
259
260 // Delete from scene and memory
261 classDiagramScene->removeItem(connection);
262 delete connection;
263 }
264
265 // Get corresponding class (entity)
266 Class *classEntity = classNode->getClassEntity();
267
268 // Delete from scene and memory
269 classDiagramScene->removeItem(classNode);
270 delete classNode;
271
272 // Delete from stored classes
273 storedClasses.erase(storedClasses.find(classEntity->getName()));
274
275 // Delete class from diagram and memory
276 classDiagram.removeClass(classEntity);
277 delete classEntity;
278}

394void ClassDiagramWindow::connectNodes()
395{
396 // Create new line in scene
397 newLine->initialize(firstToSelect, secondToSelect, firstToSelect == secondToSelect);
398 classDiagramScene->addItem(newLine);
399 firstToSelect->addLine(newLine);
400 if(firstToSelect != secondToSelect)
401 secondToSelect->addLine(newLine);
402
403 // Create new relationship in class diagram
404 Relationship *relationship;
405 const std::type_info &relationshipType = typeid(*newLine);
406 if (relationshipType == typeid(AgregationLine)) {
407 relationship = new Aggregation{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
408 } else if (relationshipType == typeid(CompositionLine)) {
409 relationship = new Composition{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
410 } else if (relationshipType == typeid(DirectedAssociationLine)) {
411 relationship = new DirectedAssociation{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
412 } else if (relationshipType == typeid(GeneralizationLine)) {
413 relationship = new Generalization{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
414 } else if (relationshipType == typeid(RealizationLine)) {
415 relationship = new Realization{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
416 } else if (relationshipType == typeid(AssociationLine)) {
417 auto *associationLine = dynamic_cast<AssociationLine *>(newLine);
418
419 relationship = new UndirectedAssociation{
420 firstToSelect->getClassEntity(),
421 secondToSelect->getClassEntity(),
422 associationLine->getName().toStdString(),
423 associationLine->getFirstCardinality().toStdString(),
424 associationLine->getSecondCardinality().toStdString()
425 };
426 } else {
427 throw std::logic_error{"Unknown type of the line"};
428 }
429
430 classDiagram.addRelationship(relationship);
431 storedRelationships.insert({newLine, relationship});
432
433 firstToSelect = nullptr;
434 secondToSelect = nullptr;
435 newLine = nullptr;
436
437 sceneUpdateObservable->sceneChanged();
438}

475void ClassDiagramWindow::redrawClassDiagram()
476{
477 // Add classes
478 for (const auto &item: classDiagram.getClasses()) {
479 auto classNode = new ClassNode(item, &storedClasses, sceneUpdateObservable);
480
481 connect(classNode->emitor, &ClassNodeEmmitor::nodePressed, this, &ClassDiagramWindow::nodePressed);
482 classDiagramScene->addItem(classNode);
483 storedClasses.insert({item->getName(), classNode});
484 }
485
486 // Add relationships
487 for (auto item: classDiagram.getRelationships()) {
488 std::string firstClassName{item->getFirstClass()->getName()};
489 std::string secondClassName{item->getSecondClass()->getName()};
490
491 ClassNode *firstClassNode = storedClasses.find(firstClassName)->second;
492 ClassNode *secondClassNode = storedClasses.find(secondClassName)->second;
493
494 Line *line;
495 const std::type_info &relationshipType = typeid(*item);
496 if (relationshipType == typeid(Aggregation)) {
497 line = new AgregationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
498 } else if (relationshipType == typeid(Composition)) {
499 line = new CompositionLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
500 } else if (relationshipType == typeid(DirectedAssociation)) {
501 line = new DirectedAssociationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
502 } else if (relationshipType == typeid(Generalization)) {
503 line = new GeneralizationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
504 } else if (relationshipType == typeid(Realization)) {
505 line = new RealizationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
506 } else if (relationshipType == typeid(UndirectedAssociation)) {
507 line = new AssociationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
508 auto *associationLine = dynamic_cast<AssociationLine *>(line);
509
510 // Add cardinalities
511 auto *undirectedAssociation = dynamic_cast<UndirectedAssociation *>(item);
512 auto firstCardinality = QString::fromStdString(undirectedAssociation->getFirstClassCardinality());
513 auto secondCardinality = QString::fromStdString(undirectedAssociation->getSecondClassCardinality());
514
515 associationLine->setFirstCardinality(firstCardinality);
516 associationLine->setSecondCardinality(secondCardinality);
517 } else {
518 throw std::logic_error{"Unknown relationship"};
519 }
520
521 // Add to stored relationships
522 storedRelationships.insert({line, item});
523
524 // Add to scene
525 line->initialize(firstClassNode, secondClassNode, firstClassNode == secondClassNode);
526 classDiagramScene->addItem(line);
527 firstClassNode->addLine(line);
528 secondClassNode->addLine(line);
529 }
530}

535void ClassDiagramWindow::openButtonClicked()
536{
537 if (!classDiagramScene->items().isEmpty() && !isSaved) {
538 auto clickedButton = QMessageBox::question(
539 this,
540 "Class diagram opening collision",
541 "You are trying to open new class diagram before saving the edited one. Are you sure to continue?"

543 );
544
545 if (clickedButton == QMessageBox::No) {
546 // Stop here, nothing will be loaded

550

553 "Choose diagram to open",
554 "",
555 "XML files (*.xml)",
556 nullptr,
557 QFileDialog::HideNameFilterDetails
558 );
559

561 ClassDiagram loadedDiagram = classDiagramManager->loadDiagram(file.toStdString());
562
563 // Remember source file

187void ClassDiagramWindow::removeClassNode()
188{
189 QList<QGraphicsItem *> selectedItems = classDiagramScene->selectedItems();
190 for (QGraphicsItem *item : selectedItems)
191 {
192 ClassNode *node = dynamic_cast<ClassNode *>(item);
193 if(node)
194 {
195 QSet<Line *> connections = node->getConnections();
196 for(Line * connection : connections)
197 delete connection;
198
199 classDiagramScene->removeItem(item);
200 delete item;
201 }
202 }

311ClassNode *ClassDiagramWindow::getSelectedNode()
312{
313 ClassNode *toReturn;
314 QList<QGraphicsItem *> selectedNodes = classDiagramScene->selectedItems();
315 if(selectedNodes.count() > 1)
316 {

318 {

201void ClassDiagramWindow::addClassNode()
202{
203 // Numbering and positioning unique names
204 static unsigned int classNumber = 1;
205 static const unsigned int nodeSpace = 50;
206
207 // Create new class in class diagram
208 Class *newClass = new Class{"New class " + std::to_string(classNumber++), std::make_tuple(0, 0)};
209
210 classDiagram.addClass(newClass);
211
212 // Create new GUI class node

214 newClassNode->setPos((newClassNode->boundingRect().width() + nodeSpace) * (classNumber % 5),
215 (newClassNode->boundingRect().height() + nodeSpace) * (classNumber / 5));

219 storedClasses.insert({newClass->getName(), newClassNode});
220

245void ClassDiagramWindow::removeClassNode(ClassNode *classNode)
246{
247 QVector<Line *> connections = classNode->getConnections();
248 for(Line *connection: connections) {
249 // Delete corresponding relationship from class diagram and memory
250 Relationship *relationship = storedRelationships.find(connection)->second;
251
252 classDiagram.removeRelationship(relationship);
253 delete relationship;
254
255 // Delete from stored relationships
256 storedRelationships.erase(connection);
257
258 // Delete from scene and memory
259 classDiagramScene->removeItem(connection);
260 delete connection;
261 }
262
263 // Get corresponding class (entity)
264 Class *classEntity = classNode->getClassEntity();
265
266 // Delete from scene and memory
267 classDiagramScene->removeItem(classNode);
268 delete classNode;
269
270 // Delete from stored classes
271 storedClasses.erase(storedClasses.find(classEntity->getName()));
272
273 // Delete class from diagram and memory
274 classDiagram.removeClass(classEntity);
275 delete classEntity;
276}

373void ClassDiagramWindow::connectNodes()
374{
375 // Create new line in scene
376 newLine->initialize(firstToSelect, secondToSelect, firstToSelect == secondToSelect);
377 classDiagramScene->addItem(newLine);
378 firstToSelect->addLine(newLine);
379 if(firstToSelect != secondToSelect)
380 secondToSelect->addLine(newLine);
381
382 // Create new relationship in class diagram
383 Relationship *relationship;
384 const std::type_info &relationshipType = typeid(*newLine);
385 if (relationshipType == typeid(AgregationLine)) {
386 relationship = new Aggregation{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
387 } else if (relationshipType == typeid(CompositionLine)) {
388 relationship = new Composition{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
389 } else if (relationshipType == typeid(DirectedAssociationLine)) {
390 relationship = new DirectedAssociation{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
391 } else if (relationshipType == typeid(GeneralizationLine)) {
392 relationship = new Generalization{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
393 } else if (relationshipType == typeid(RealizationLine)) {
394 relationship = new Realization{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
395 } else if (relationshipType == typeid(AssociationLine)) {
396 auto *associationLine = dynamic_cast<AssociationLine *>(newLine);
397
398 relationship = new UndirectedAssociation{
399 firstToSelect->getClassEntity(),
400 secondToSelect->getClassEntity(),
401 associationLine->getName().toStdString(),
402 associationLine->getFirstCardinality().toStdString(),
403 associationLine->getSecondCardinality().toStdString()
404 };
405 } else {
406 throw std::logic_error{"Unknown type of the line"};
407 }
408
409 classDiagram.addRelationship(relationship);
410 storedRelationships.insert({newLine, relationship});
411
412 firstToSelect = nullptr;
413 secondToSelect = nullptr;
414 newLine = nullptr;
415
416 sceneUpdateObservable->sceneChanged();
417}

517void ClassDiagramWindow::redrawClassDiagram()
518{
519 // Add classes
520 for (const auto &item: classDiagram.getClasses()) {
521 auto classNode = new ClassNode(item, &storedClasses, sceneUpdateObservable);
522
523 connect(&(classNode->emitter), &ClassNodeEmitter::nodePressed, this, &ClassDiagramWindow::nodePressed);
524 classDiagramScene->addItem(classNode);
525 storedClasses.insert({item->getName(), classNode});
526 }
527
528 // Add relationships
529 for (auto item: classDiagram.getRelationships()) {
530 std::string firstClassName{item->getFirstClass()->getName()};
531 std::string secondClassName{item->getSecondClass()->getName()};
532
533 ClassNode *firstClassNode = storedClasses.find(firstClassName)->second;
534 ClassNode *secondClassNode = storedClasses.find(secondClassName)->second;
535
536 Line *line;
537 const std::type_info &relationshipType = typeid(*item);
538 if (relationshipType == typeid(Aggregation)) {
539 line = new AgregationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
540 } else if (relationshipType == typeid(Composition)) {
541 line = new CompositionLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
542 } else if (relationshipType == typeid(DirectedAssociation)) {
543 line = new DirectedAssociationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
544 } else if (relationshipType == typeid(Generalization)) {
545 line = new GeneralizationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
546 } else if (relationshipType == typeid(Realization)) {
547 line = new RealizationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
548 } else if (relationshipType == typeid(UndirectedAssociation)) {
549 line = new AssociationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
550 auto *associationLine = dynamic_cast<AssociationLine *>(line);
551
552 // Add cardinalities
553 auto *undirectedAssociation = dynamic_cast<UndirectedAssociation *>(item);
554 auto firstCardinality = QString::fromStdString(undirectedAssociation->getFirstClassCardinality());
555 auto secondCardinality = QString::fromStdString(undirectedAssociation->getSecondClassCardinality());
556
557 associationLine->setFirstCardinality(firstCardinality);
558 associationLine->setSecondCardinality(secondCardinality);
559 } else {
560 throw std::logic_error{"Unknown relationship"};
561 }
562
563 // Add to stored relationships
564 storedRelationships.insert({line, item});
565
566 // Add to scene
567 line->initialize(firstClassNode, secondClassNode, firstClassNode == secondClassNode);
568 classDiagramScene->addItem(line);
569 firstClassNode->addLine(line);
570 secondClassNode->addLine(line);
571 }
572}

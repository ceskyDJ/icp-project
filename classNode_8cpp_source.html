164QRectF ClassNode::getWholeRect(std::vector<QString> &attributePrintable,
165 std::vector<QString> &methodPrintable) const

178 + nameRect.height() * methodPrintable.size());
179 wholeRect.adjust(0, -Padding, 0, Padding);

181 wholeRect.setHeight(wholeRect.height() + nameRect.height() );

395bool ClassNode::isMethodInherited(QString methodName) const
396{
397 QVector<ClassNode *> generalisations(QVector<ClassNode *>(0));
398 for (Line * relationship: connectedLines)
399 {
400 Line *temp = dynamic_cast<GeneralizationLine *>(relationship);
401 if(!temp)
402 temp = dynamic_cast<RealizationLine *>(relationship);
403 if(temp && temp->getToClassNode() != this)
404 generalisations.push_back(temp->getToClassNode());
405 }
406
407 for(ClassNode *node: generalisations)
408 {
409 if (node->getClassEntity()->findMethodByName(methodName.toStdString()) != nullptr)
410 return true;
411 }
412
413 return false;
414}

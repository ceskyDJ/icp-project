136QRectF ClassNode::getWholeRect(std::vector<QString> attributePrintable,
137 std::vector<QString> methodPrintable) const
138{
139 QFontMetricsF metrics{qApp->font()};
140 QRectF nameRect = metrics.boundingRect(QString::fromStdString(classEntity.getName()));
141
142 int maxWidth = nameRect.width();
143 getMaxWidth(attributePrintable, &maxWidth);
144 getMaxWidth(methodPrintable, &maxWidth);
145 maxWidth += Padding * 2;
146
147 QRectF wholeRect = nameRect;
148 wholeRect.setWidth(maxWidth);
149 wholeRect.setHeight(nameRect.height() + 2 * lineIndent + nameRect.height() * attributePrintable.size()
150 + nameRect.height() * methodPrintable.size());
151 wholeRect.adjust(0, -Padding, 0, Padding);
152 if(classEntity.getClassType() == (ClassType)ClassType::INTERFACE)
153 wholeRect.setHeight(wholeRect.height() + nameRect.height() );
154 return wholeRect;
155}

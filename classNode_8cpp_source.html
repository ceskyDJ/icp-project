161QRectF ClassNode::getWholeRect(std::vector<QString> &attributePrintable,
162 std::vector<QString> &methodPrintable) const
163{
164 QFontMetricsF metrics{qApp->font()};
165 QRectF nameRect = metrics.boundingRect(QString::fromStdString(classEntity->getName()));
166
167 int maxWidth = nameRect.width();
168 getMaxWidth(attributePrintable, &maxWidth);
169 getMaxWidth(methodPrintable, &maxWidth);
170 maxWidth += Padding * 2;
171
172 QRectF wholeRect = nameRect;
173 wholeRect.setWidth(maxWidth);
174 wholeRect.setHeight(nameRect.height() + 2 * lineIndent + nameRect.height() * attributePrintable.size()
175 + nameRect.height() * methodPrintable.size());
176 wholeRect.adjust(0, -Padding, 0, Padding);
177 if(classEntity->getClassType() == (ClassType)ClassType::INTERFACE)
178 wholeRect.setHeight(wholeRect.height() + nameRect.height() );
179 return wholeRect;

288void ClassNode::mouseDoubleClickEvent(QGraphicsSceneMouseEvent */*event*/)
289{
290 Class newClassEntity{*classEntity};

292

294 // Propagate name change to existing classes map

296 // Delete old key with this class node

298 existingClasses->erase(itemForDeletion);

408bool ClassNode::isMethodInherited(QString methodName) const
409{
410 QVector<ClassNode *> generalisations(QVector<ClassNode *>(0));
411 for (Line * relationship: connectedLines)
412 {
413 GeneralizationLine *temp = (GeneralizationLine *)relationship;
414 if(temp && temp->getToClassNode() != this)
415 generalisations.push_back(temp->getToClassNode());
416 }
417
418 for(ClassNode *node: generalisations)
419 {
420 if (node->getClassEntity()->findMethodByName(methodName.toStdString()) != nullptr)
421 return true;
422 }
423
424 return false;

65 prepareGeometryChange();

105QRectF MessageLine::boundingRect() const
106{
107 static qreal textHeight = getTextBoundingBox("").height();
108 QRectF fromLifebox = getObjectLifeBox(fromObject);
109 QRectF toLifebox = getObjectLifeBox(toObject);
110
111 qreal yPadding = ActivationGraphicsObjectBase::lifeSpaceStart +
112 ActivationGraphicsObjectBase::headerHeight;
113 qreal maxLifeboxHeight = ActivationGraphicsObjectBase::height - yPadding;
114
115 orderRects(&fromLifebox, &toLifebox);
116 QRectF toRet = QRectF{
117 fromLifebox.x() + ActivationGraphicsObjectBase::lifeboxWidth,
118 yPadding + maxLifeboxHeight * message->getSendingTime(),
119 toLifebox.x() - fromLifebox.x() - ActivationGraphicsObjectBase::lifeboxWidth,
120 arrowHeight + textHeight * 1.5 //padded because text should be up on the line
121 };
122
123 //move rectangle up so the line will lie on bottom edge of this rectangle + arrowHeight
124 toRet.translate(0, - toRet.height() + arrowHeight);
125 return toRet;
126}

270void MessageLine::moveLine(qreal dy, bool moveIfNotValidBefore)
271{
272 QRectF rect = boundingRect();
273
274 qreal yPadding = ActivationGraphicsObjectBase::lifeSpaceStart + ActivationGraphicsObjectBase::headerHeight;
275 qreal maxBoxHeight = ActivationGraphicsObjectBase::height - yPadding;
276
277 qreal newSendingTime = (rect.y() + rect.height() - arrowHeight + dy - yPadding) / maxBoxHeight;
278 //change position if sending time is not outside of interval <0.1> and if new position is inside of lifeboxes
279 //or if previous position is not inside of lifeboxes or if creaated flag is true
280 if((newSendingTime >= 0 && newSendingTime <= 1) && (isInObjectsLifeBox(newSendingTime) ||
281 (!isInObjectsLifeBox(message->getSendingTime()) && moveIfNotValidBefore) || createFlag || destroyFlag))
282 {
283 if(destroyFlag) // after lifebox a remove cross wil be drawn
284 {
285 if (toObject->getStartOfLifeBox() < newSendingTime) //new seding time has to be smaller than lifbox start
286 toObject->setLifeEndDestroy(newSendingTime);
287 else
288 return;
289 }
290 if (createFlag) // means that method is not called externally
291 toObject->setLifeStart(newSendingTime);
292
293 message->setSendingTime(newSendingTime);
294 }
295}

109 painter->setPen(usedPen);
110
111 if(fromObject->x() > toObject->x()) //arrow goes from right to left
112 line.setLine(line.x2(), line.y2(), line.x1(), line.y2());
113
114 line.setLength(line.length() - arrowWidth); //make space for arrow
115 painter->drawLine(line);
116
117 //set painter for drawing arrow - translate, rotate and set pen

119 painter->translate(line.p2().x(), line.y2());
120 QPen temp = painter->pen();

122 painter->setPen(temp);
123 if(fromObject->x() > toObject->x())
124 painter->rotate(180);

133QRectF MessageLine::boundingRect() const
134{
135 static qreal textHeight = getTextBoundingBox("").height();
136 QRectF fromLifebox = getObjectLifeBox(fromObject);
137 QRectF toLifebox = getObjectLifeBox(toObject);
138
139 qreal yPadding = ActivationGraphicsObjectBase::lifeSpaceStart +
140 ActivationGraphicsObjectBase::headerHeight;
141 qreal maxLifeboxHeight = ActivationGraphicsObjectBase::height - yPadding;
142
143 orderRects(&fromLifebox, &toLifebox);
144 QRectF toRet = QRectF{
145 fromLifebox.x() + ActivationGraphicsObjectBase::lifeboxWidth,
146 yPadding + maxLifeboxHeight * message->getSendingTime(),
147 toLifebox.x() - fromLifebox.x() - ActivationGraphicsObjectBase::lifeboxWidth,
148 arrowHeight + textHeight * 1.5 //padded because text should be up on the line
149 };
150
151 //move rectangle up so the line will lie on bottom edge of this rectangle + arrowHeight
152 toRet.translate(0, - toRet.height() + arrowHeight);
153 return toRet;
154}

308void MessageLine::moveLine(qreal dy, bool moveIfNotValidBefore)
309{
310 QRectF rect = boundingRect();
311
312 qreal yPadding = ActivationGraphicsObjectBase::lifeSpaceStart + ActivationGraphicsObjectBase::headerHeight;
313 qreal maxBoxHeight = ActivationGraphicsObjectBase::height - yPadding;
314
315 qreal newSendingTime = (rect.y() + rect.height() - arrowHeight + dy - yPadding) / maxBoxHeight;
316
317 // arrow is not too up or too down
318 bool inAllowedHeight = (newSendingTime >= 0 && newSendingTime <= 1);
319 //if object was not in valid position (between both lifeboxes), can this move? - It according to an argument
320 bool moveIfNotValidBeforeAllowed = (!isInObjectsLifeBox(message->getSendingTime()) && moveIfNotValidBefore);
321 //If the line wil bee moved and then object will be beteween lifeboxes, then true, else false
322 bool willBeInLifeboxes = isInObjectsLifeBox(newSendingTime);
323 if(inAllowedHeight && (willBeInLifeboxes || moveIfNotValidBeforeAllowed || createFlag || destroyFlag))
324 {
325 // If destroyFlag is true and object is under the life start, changes end of life of object
326 if(destroyFlag && toObject->getStartOfLifeBox() < newSendingTime)
327 {
328 toObject->setLifeEndDestroy(newSendingTime);
329
330 //if after move, object will not be in
331 if(!isInObjectsLifeBox(newSendingTime))
332 toObject->setLifeEndDestroy((rect.y() + rect.height() - arrowHeight + dy + 1 - yPadding) / maxBoxHeight);
333
334 }
335 if (createFlag) // means that method is not called externally
336 toObject->setLifeStart(newSendingTime);
337
338 message->setSendingTime(newSendingTime);
339 }
340}

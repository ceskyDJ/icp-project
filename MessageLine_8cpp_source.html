116QRectF MessageLine::boundingRect() const
117{
118 static qreal textHeight = getTextBoundingBox("").height();
119 QRectF fromLifebox = getObjectLifeBox(fromObject);
120 QRectF toLifebox = getObjectLifeBox(toObject);
121
122 qreal yPadding = ActivationGraphicsObjectBase::lifeSpaceStart +
123 ActivationGraphicsObjectBase::headerHeight;
124 qreal maxLifeboxHeight = ActivationGraphicsObjectBase::height - yPadding;
125
126 orderRects(&fromLifebox, &toLifebox);
127 QRectF toRet = QRectF{
128 fromLifebox.x() + ActivationGraphicsObjectBase::lifeboxWidth,
129 yPadding + maxLifeboxHeight * message->getSendingTime(),
130 toLifebox.x() - fromLifebox.x() - ActivationGraphicsObjectBase::lifeboxWidth,
131 arrowHeight + textHeight * 1.5 //padded because text should be up on the line
132 };
133
134 //move rectangle up so the line will lie on bottom edge of this rectangle + arrowHeight
135 toRet.translate(0, - toRet.height() + arrowHeight);

281void MessageLine::moveLine(qreal dy, bool moveIfNotValidBefore)
282{
283 QRectF rect = boundingRect();
284
285 qreal yPadding = ActivationGraphicsObjectBase::lifeSpaceStart + ActivationGraphicsObjectBase::headerHeight;
286 qreal maxBoxHeight = ActivationGraphicsObjectBase::height - yPadding;
287
288 qreal newSendingTime = (rect.y() + rect.height() - arrowHeight + dy - yPadding) / maxBoxHeight;
289
290 // arrow is not too up or too down
291 bool inAllowedHeight = (newSendingTime >= 0 && newSendingTime <= 1);
292 //if object was not in valid position (between both lifeboxes), can this move? - It according to an argument
293 bool moveIfNotValidBeforeAllowed = (!isInObjectsLifeBox(message->getSendingTime()) && moveIfNotValidBefore);
294 //If the line wil bee moved and then object will be beteween lifeboxes, then true, else false
295 bool willBeInLifeboxes = isInObjectsLifeBox(newSendingTime);
296 if(inAllowedHeight && (willBeInLifeboxes || moveIfNotValidBeforeAllowed || createFlag || destroyFlag))
297 {
298 // If destroyFlag is true and object is under the life start, changes end of life of object
299 if(destroyFlag && toObject->getStartOfLifeBox() < newSendingTime)
300 {
301 toObject->setLifeEndDestroy(newSendingTime);
302
303 //if after move, object will not be in
304 if(!isInObjectsLifeBox(newSendingTime))
305 toObject->setLifeEndDestroy((rect.y() + rect.height() - arrowHeight + dy + 1 - yPadding) / maxBoxHeight);
306
307 }
308 if (createFlag) // means that method is not called externally
309 toObject->setLifeStart(newSendingTime);
310
311 message->setSendingTime(newSendingTime);
312 }
313}

25SequenceDiagram FileSequenceDiagramRepository::loadDiagram()
26{
27 QFile file{fileName.c_str()};
28
29 if (fileName.empty()) {
30 throw InvalidDataStorageException{"No source file set"};
31 }
32
33 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
34 file.close();
35 throw InvalidDataStorageException{"Cannot open source file"};
36 }
37
38 // Load data from XML file
39 QDomDocument xml{};
40 xml.setContent(&file);
41 file.close();
42
43 QDomElement rootElement{xml.documentElement()};
44 if (rootElement.tagName() != "sequence-diagram") {
45 throw InvalidInputDataException{"Invalid root element " + rootElement.tagName().toStdString()
46 + R"(, "sequence-diagram" expected)"};
47 }
48
49 // Prepare empty ClassDiagram
50 SequenceDiagram sequenceDiagram{};
51
52 // Process actors
53 QDomNodeList actorsContainerSearch{rootElement.elementsByTagName("actors")};
54 if (actorsContainerSearch.size() != 1 || !actorsContainerSearch.item(0).isElement()) {
55 throw InvalidInputDataException{R"(Element "sequence-diagram" must contain exactly 1 element "actors")"};
56 }
57
58 QDomElement actorsContainer{actorsContainerSearch.item(0).toElement()};
59 QDomNodeList actors{actorsContainer.elementsByTagName("actor")};
60
61 QDomElement xmlActor{actors.item(0).toElement()};
62 while (!xmlActor.isNull()) {
63 // Name
64 if (xmlActor.text().isEmpty()) {
65 throw InvalidInputDataException{R"(Element "actor" must contain actor's name)"};
66 }
67
68 Actor objActor{xmlActor.text().toStdString()};
69 sequenceDiagram.addActor(objActor);
70
71 xmlActor = xmlActor.nextSiblingElement("actor");
72 }
73
74 // Process objects
75 QDomNodeList objectsContainerSearch{rootElement.elementsByTagName("objects")};
76 if (objectsContainerSearch.size() != 1 || !objectsContainerSearch.item(0).isElement()) {
77 throw InvalidInputDataException{R"(Element "sequence-diagram" must contain exactly 1 element "objects")"};
78 }
79
80 QDomElement objectsContainer{objectsContainerSearch.item(0).toElement()};
81 QDomNodeList objects{objectsContainer.elementsByTagName("object")};
82
83 QDomElement xmlObject{objects.item(0).toElement()};
84 while (!xmlObject.isNull()) {
85 sequenceDiagram.addObject(loadObject(xmlObject));
86
87 xmlObject = xmlObject.nextSiblingElement("object");
88 }
89
90 // Process messages
91 QDomNodeList messagesContainerSearch{rootElement.elementsByTagName("messages")};
92 if (messagesContainerSearch.size() != 1 || !messagesContainerSearch.item(0).isElement()) {
93 throw InvalidInputDataException{R"(Element "sequence-diagram" must contain exactly 1 element "messages")"};
94 }
95
96 QDomElement messagesContainer{messagesContainerSearch.item(0).toElement()};
97 QDomNodeList messages{messagesContainer.elementsByTagName("message")};
98
99 QDomElement xmlMessage{messages.item(0).toElement()};
100 while (!xmlMessage.isNull()) {
101 sequenceDiagram.addMessage(loadMessage(xmlMessage, sequenceDiagram));
102
103 xmlMessage = xmlMessage.nextSiblingElement("message");
104 }
105
106 return sequenceDiagram;
107}

115void FileSequenceDiagramRepository::saveDiagram(SequenceDiagram diagram)
116{
117 if (fileName.empty()) {
118 throw InvalidDataStorageException{"No target file set"};
119 }
120
121 // Convert objects to XML
122 QDomDocument xml{};
123
124 QDomElement rootElement{xml.createElement("sequence-diagram")};
125 xml.appendChild(rootElement);
126
127 // Actors
128 QDomElement actorsContainer{xml.createElement("actors")};
129 for (const Actor &actor: diagram.getActors()) {
130 QDomElement xmlActor{xml.createElement("actor")};
131
132 QDomText actorName{xml.createTextNode(actor.getName().c_str())};
133 xmlActor.appendChild(actorName);
134
135 actorsContainer.appendChild(xmlActor);
136 }
137
138 // Objects
139 QDomElement objectsContainer{xml.createElement("objects")};
140 for (const Object &object: diagram.getObjects()) {
141 QDomElement xmlObject{xml.createElement("object")};
142 xmlObject.setAttribute("instance-class", object.getInstanceClass()->getName().c_str());
143 xmlObject.setAttribute("life-start", object.getLifeStart());
144 xmlObject.setAttribute("life-length", object.getLifeLength());
145
146 QDomText objectName{xml.createTextNode(object.getName().c_str())};
147 xmlObject.appendChild(objectName);
148
149 objectsContainer.appendChild(xmlObject);
150 }
151
152 // Messages
153 QDomElement messagesContainer{xml.createElement("messages")};
154 for (const Message &message: diagram.getMessages()) {
155 QDomElement xmlMessage{xml.createElement("message")};
156 xmlMessage.setAttribute("name", message.getMethod()->getName().c_str());
157 xmlMessage.setAttribute("type", message.getType().serialize().c_str());
158 xmlMessage.setAttribute("sending-time", message.getSendingTime());
159
160 // Message sender
161 QDomElement xmlMessageSender{xml.createElement("sender")};
162
163 Message::MessageSender *messageSender{message.getMessageSender()};
164 if (typeid(*messageSender) == typeid(Actor)) {
165 xmlMessageSender.setAttribute("type", "ACTOR");
166 } else {
167 // typeid(*messageSender) == typeid(Object)
168 xmlMessageSender.setAttribute("type", "OBJECT");
169 }
170
171 QDomText messageSenderName{xml.createTextNode(messageSender->getName().c_str())};
172 xmlMessageSender.appendChild(messageSenderName);
173
174 xmlMessage.appendChild(xmlMessageSender);
175
176 // Message recipient
177 QDomElement xmlMessageRecipient{xml.createElement("recipient")};
178
179 Message::MessageRecipient *messageRecipient{message.getMessageRecipient()};
180 if (typeid(*messageRecipient) == typeid(Actor)) {
181 xmlMessageRecipient.setAttribute("type", "ACTOR");
182 } else {
183 // typeid(*messageRecipient) == typeid(Object)
184 xmlMessageRecipient.setAttribute("type", "OBJECT");
185 }
186
187 QDomText messageRecipientName{xml.createTextNode(messageRecipient->getName().c_str())};
188 xmlMessageRecipient.appendChild(messageRecipientName);
189
190 messagesContainer.appendChild(xmlMessage);
191 }
192
193 rootElement.appendChild(actorsContainer);
194 rootElement.appendChild(objectsContainer);
195 rootElement.appendChild(messagesContainer);
196
197 // Save data to XML file
198 QFile file{fileName.c_str()};
199 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
200 file.close();
201 throw InvalidDataStorageException{"Cannot open target file"};
202 }
203
204 QTextStream xmlStream{&file};
205 xmlStream << xml.toString(4);
206 file.close();
207}

216Object FileSequenceDiagramRepository::loadObject(QDomElement &xmlObject)
217{
218 // Instance class
219 if (!xmlObject.hasAttribute("instance-class")) {
220 throw InvalidInputDataException{R"(Element "object" must have mandatory attribute "instance-class")"};
221 }
222
223 ClassReference instanceClass{xmlObject.attribute("instance-class").toStdString()};
224 try {
225 instanceClass = classDiagram.findClassByName(instanceClass.getReferredClassName());
226 } catch (std::invalid_argument &e) {}
227
228 // Life start
229 if (!xmlObject.hasAttribute("life-start")) {
230 throw InvalidInputDataException{R"(Element "object" must have mandatory attribute "life-start")"};
231 }
232
233 bool lifeStartConvOk;
234 double lifeStart = xmlObject.attribute("life-start").toDouble(&lifeStartConvOk);
235 if (!lifeStartConvOk) {
236 throw InvalidInputDataException{R"(Value of attribute "life-start" must be valid decimal)"};
237 }
238
239 // Life end
240 if (!xmlObject.hasAttribute("life-length")) {
241 throw InvalidInputDataException{R"(Element "object" must have mandatory attribute "life-length")"};
242 }
243
244 bool lifeLengthConvOk;
245 double lifeLength = xmlObject.attribute("life-length").toDouble(&lifeLengthConvOk);
246 if (!lifeLengthConvOk) {
247 throw InvalidInputDataException{R"(Value of attribute "life-length" must be valid decimal)"};
248 }
249
250 // Name
251 if (xmlObject.text().isEmpty()) {
252 throw InvalidInputDataException{R"(Element "object" must contain object name)"};
253 }
254
255 return Object{instanceClass, lifeStart, lifeLength, xmlObject.text().toStdString()};

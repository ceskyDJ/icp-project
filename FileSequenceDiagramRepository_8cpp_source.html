49 // Prepare empty ClassDiagram

51
52 // Process actors
53 QDomNodeList actorsContainerSearch{rootElement.elementsByTagName("actors")};

55 throw InvalidInputDataException{R"(Element "sequence-diagram" must contain exactly 1 element "actors")"};
56 }

59 QDomNodeList actors{actorsContainer.elementsByTagName("actor")};
60
61 QDomElement xmlActor{actors.item(0).toElement()};

73 }

75 sequenceDiagram.addActor(objActor);
76
77 xmlActor = xmlActor.nextSiblingElement("actor");
78 }
79
80 // Process objects

82 if (objectsContainerSearch.size() != 1 || !objectsContainerSearch.item(0).isElement()) {
83 throw InvalidInputDataException{R"(Element "sequence-diagram" must contain exactly 1 element "objects")"};
84 }

87 QDomNodeList objects{objectsContainer.elementsByTagName("object")};

101
102 // Process messages

109 QDomNodeList messages{messagesContainer.elementsByTagName("message")};

127void FileSequenceDiagramRepository::saveDiagram(const SequenceDiagram &diagram)
128{
129 if (storageName.empty()) {
130 throw InvalidDataStorageException{"No target file set"};
131 }
132
133 // Convert objects to XML
134 QDomDocument xml{};
135
136 QDomElement rootElement{xml.createElement("sequence-diagram")};
137 xml.appendChild(rootElement);
138
139 // Actors
140 QDomElement actorsContainer{xml.createElement("actors")};
141 for (const Actor *actor: diagram.getActors()) {
142 QDomElement xmlActor{xml.createElement("actor")};
143
144 QDomText actorName{xml.createTextNode(actor->getName().c_str())};
145 xmlActor.appendChild(actorName);
146
147 actorsContainer.appendChild(xmlActor);
148 }
149
150 // Objects
151 QDomElement objectsContainer{xml.createElement("objects")};
152 for (const Object *object: diagram.getObjects()) {
153 QDomElement xmlObject{xml.createElement("object")};
154 xmlObject.setAttribute("instance-class", object->getInstanceClass()->getName().c_str());
155 xmlObject.setAttribute("life-start", object->getLifeStart());
156 xmlObject.setAttribute("life-length", object->getLifeLength());
157
158 QDomText objectName{xml.createTextNode(object->getName().c_str())};
159 xmlObject.appendChild(objectName);
160
161 objectsContainer.appendChild(xmlObject);
162 }
163
164 // Messages
165 QDomElement messagesContainer{xml.createElement("messages")};
166 for (const Message *message: diagram.getMessages()) {
167 QDomElement xmlMessage{xml.createElement("message")};
168 xmlMessage.setAttribute("name", message->getMethod()->getName().c_str());
169 xmlMessage.setAttribute("type", message->getType().serialize().c_str());
170 xmlMessage.setAttribute("sending-time", message->getSendingTime());
171
172 // Message sender
173 QDomElement xmlMessageSender{xml.createElement("sender")};
174
175 Message::MessageSender *messageSender{message->getMessageSender()};
176 if (typeid(*messageSender) == typeid(Actor)) {
177 xmlMessageSender.setAttribute("type", "ACTOR");
178 } else {
179 // typeid(*messageSender) == typeid(Object)
180 xmlMessageSender.setAttribute("type", "OBJECT");
181 }
182
183 QDomText messageSenderName{xml.createTextNode(messageSender->getName().c_str())};
184 xmlMessageSender.appendChild(messageSenderName);
185
186 // Message recipient
187 QDomElement xmlMessageRecipient{xml.createElement("recipient")};
188
189 Message::MessageRecipient *messageRecipient{message->getMessageRecipient()};
190 if (typeid(*messageRecipient) == typeid(Actor)) {
191 xmlMessageRecipient.setAttribute("type", "ACTOR");
192 } else {
193 // typeid(*messageRecipient) == typeid(Object)
194 xmlMessageRecipient.setAttribute("type", "OBJECT");
195 }
196
197 QDomText messageRecipientName{xml.createTextNode(messageRecipient->getName().c_str())};
198 xmlMessageRecipient.appendChild(messageRecipientName);
199
200 xmlMessage.appendChild(xmlMessageSender);
201 xmlMessage.appendChild(xmlMessageRecipient);
202 messagesContainer.appendChild(xmlMessage);
203 }
204
205 rootElement.appendChild(actorsContainer);
206 rootElement.appendChild(objectsContainer);
207 rootElement.appendChild(messagesContainer);
208
209 // Save data to XML file
210 QFile file{storageName.c_str()};
211 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
212 file.close();
213 throw InvalidDataStorageException{"Cannot open target file"};
214 }
215
216 QTextStream xmlStream{&file};
217 xmlStream << xml.toString(4);
218 file.close();
219}

228Object *FileSequenceDiagramRepository::loadObject(QDomElement &xmlObject)
229{
230 // Instance class
231 if (!xmlObject.hasAttribute("instance-class")) {
232 throw InvalidInputDataException{R"(Element "object" must have mandatory attribute "instance-class")"};
233 }
234
235 ClassReference instanceClass{xmlObject.attribute("instance-class").toStdString()};
236 Class *searchedClass = classDiagram->findClassByName(instanceClass.getReferredClassName());
237 if (searchedClass != nullptr) {
238 instanceClass.storePointer(searchedClass);
239 }
240
241 // Life start
242 if (!xmlObject.hasAttribute("life-start")) {
243 throw InvalidInputDataException{R"(Element "object" must have mandatory attribute "life-start")"};
244 }
245
246 bool lifeStartConvOk;
247 double lifeStart = xmlObject.attribute("life-start").toDouble(&lifeStartConvOk);
248 if (!lifeStartConvOk) {
249 throw InvalidInputDataException{R"(Value of attribute "life-start" must be valid decimal)"};
250 }
251
252 // Life end
253 if (!xmlObject.hasAttribute("life-length")) {
254 throw InvalidInputDataException{R"(Element "object" must have mandatory attribute "life-length")"};
255 }
256
257 bool lifeLengthConvOk;
258 double lifeLength = xmlObject.attribute("life-length").toDouble(&lifeLengthConvOk);
259 if (!lifeLengthConvOk) {
260 throw InvalidInputDataException{R"(Value of attribute "life-length" must be valid decimal)"};
261 }
262
263 // Name
264 if (xmlObject.text().isEmpty()) {
265 throw InvalidInputDataException{R"(Element "object" must contain object name)"};
266 }
267
268 return new Object{instanceClass, lifeStart, lifeLength, xmlObject.text().toStdString()};

130void ClassDiagramScene::addClassNode()
131{
132 // Numbering and positioning unique names
133 static unsigned int classNumber = 1;
134 static const unsigned int nodeSpace = 50;
135
136 // Create new class in class diagram
137 auto newClass = new Class{"New class " + std::to_string(classNumber++), std::make_tuple(0, 0)};
138
139 classDiagram.addClass(newClass);
140
141 // Create new GUI class node
142 auto *newClassNode = new ClassNode(newClass, &storedClasses, sceneUpdateObservable);
143 newClassNode->setPos(
144 (newClassNode->boundingRect().width() + nodeSpace) * (classNumber % 5),
145 (newClassNode->boundingRect().height() + nodeSpace) * (classNumber / 5) // NOLINT(bugprone-integer-division)
146 );
147
148 connect(&(newClassNode->emitter), &ClassNodeEmitter::nodePressed, this, &ClassDiagramScene::nodePressed);
149 addItem(newClassNode);
150 storedClasses.insert({newClass->getName(), newClassNode});
151
152 sceneUpdateObservable->sceneChanged();
153}

252void ClassDiagramScene::connectNodes()
253{
254 // Create new line in scene
255 newLine->initialize(firstToSelect, secondToSelect, firstToSelect == secondToSelect);
256 addItem(newLine);
257 firstToSelect->addLine(newLine);
258 if(firstToSelect != secondToSelect)
259 secondToSelect->addLine(newLine);
260
261 // Create new relationship in class diagram
262 Relationship *relationship;
263 const std::type_info &relationshipType = typeid(*newLine);
264 if (relationshipType == typeid(AgregationLine)) {
265 relationship = new Aggregation{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
266 } else if (relationshipType == typeid(CompositionLine)) {
267 relationship = new Composition{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
268 } else if (relationshipType == typeid(DirectedAssociationLine)) {
269 auto *directedAssociationLine = dynamic_cast<DirectedAssociationLine *>(newLine);
270
271 relationship = new DirectedAssociation{
272 firstToSelect->getClassEntity(),
273 secondToSelect->getClassEntity(),
274 directedAssociationLine->getName().toStdString()
275 };
276 } else if (relationshipType == typeid(GeneralizationLine)) {
277 relationship = new Generalization{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
278 } else if (relationshipType == typeid(RealizationLine)) {
279 relationship = new Realization{firstToSelect->getClassEntity(), secondToSelect->getClassEntity()};
280 } else if (relationshipType == typeid(AssociationLine)) {
281 auto *associationLine = dynamic_cast<AssociationLine *>(newLine);
282
283 relationship = new UndirectedAssociation{
284 firstToSelect->getClassEntity(),
285 secondToSelect->getClassEntity(),
286 associationLine->getName().toStdString(),
287 associationLine->getFirstCardinality().toStdString(),
288 associationLine->getSecondCardinality().toStdString()
289 };
290 } else {
291 throw std::logic_error{"Unknown type of the line"};
292 }
293
294 classDiagram.addRelationship(relationship);
295 storedRelationships.insert({newLine, relationship});
296
297 firstToSelect = nullptr;
298 secondToSelect = nullptr;
299 newLine = nullptr;
300
301 sceneUpdateObservable->sceneChanged();
302}

332void ClassDiagramScene::removeClassNode(ClassNode *classNode)
333{
334 // One connection could have the same class node at its both ends
335 // One line (connection) could be deleted only once, so these class nodes'
336 // connections must be filtered for duplicates (set contains only unique items)
337 QVector<Line *> allConnections = classNode->getConnections();
338 QSet<Line *> uniqueConnections{allConnections.begin(), allConnections.end()};
339 for(Line *connection: uniqueConnections) {
340 // Delete corresponding relationship from class diagram and memory
341 Relationship *relationship = storedRelationships.find(connection)->second;
342
343 classDiagram.removeRelationship(relationship);
344 delete relationship;
345
346 // Delete from stored relationships
347 storedRelationships.erase(connection);
348
349 // Delete from scene and memory
350 removeItem(connection);
351 delete connection;
352 }
353
354 // Get corresponding class (entity)
355 Class *classEntity = classNode->getClassEntity();
356
357 // Delete from scene and memory
358 removeItem(classNode);
359 delete classNode;
360
361 // Delete from stored classes
362 storedClasses.erase(storedClasses.find(classEntity->getName()));
363
364 // Delete class from diagram and memory
365 classDiagram.removeClass(classEntity);
366 delete classEntity;
367}

393void ClassDiagramScene::redrawClassDiagram()
394{
395 // Add classes
396 for (const auto &item: classDiagram.getClasses()) {
397 auto classNode = new ClassNode(item, &storedClasses, sceneUpdateObservable);
398
399 connect(&(classNode->emitter), &ClassNodeEmitter::nodePressed, this, &ClassDiagramScene::nodePressed);
400 addItem(classNode);
401 storedClasses.insert({item->getName(), classNode});
402 }
403
404 // Add relationships
405 for (auto item: classDiagram.getRelationships()) {
406 std::string firstClassName{item->getFirstClass()->getName()};
407 std::string secondClassName{item->getSecondClass()->getName()};
408
409 ClassNode *firstClassNode = storedClasses.find(firstClassName)->second;
410 ClassNode *secondClassNode = storedClasses.find(secondClassName)->second;
411
412 Line *line;
413 const std::type_info &relationshipType = typeid(*item);
414 if (relationshipType == typeid(Aggregation)) {
415 line = new AgregationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
416 } else if (relationshipType == typeid(Composition)) {
417 line = new CompositionLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
418 } else if (relationshipType == typeid(DirectedAssociation)) {
419 line = new DirectedAssociationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
420 auto *directedAssociationLine = dynamic_cast<DirectedAssociationLine *>(line);
421
422 // Add name
423 auto name = QString::fromStdString(item->getName());
424
425 directedAssociationLine->setName(name);
426 } else if (relationshipType == typeid(Generalization)) {
427 line = new GeneralizationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
428 } else if (relationshipType == typeid(Realization)) {
429 line = new RealizationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
430 } else if (relationshipType == typeid(UndirectedAssociation)) {
431 line = new AssociationLine{&storedRelationships, &classDiagram, sceneUpdateObservable};
432 auto *associationLine = dynamic_cast<AssociationLine *>(line);
433
434 // Add name and cardinalities
435 auto *undirectedAssociation = dynamic_cast<UndirectedAssociation *>(item);
436 auto name = QString::fromStdString(undirectedAssociation->getName());
437 auto firstCardinality = QString::fromStdString(undirectedAssociation->getFirstClassCardinality());
438 auto secondCardinality = QString::fromStdString(undirectedAssociation->getSecondClassCardinality());
439
440 associationLine->setName(name);
441 associationLine->setFirstCardinality(firstCardinality);
442 associationLine->setSecondCardinality(secondCardinality);
443 } else {
444 throw std::logic_error{"Unknown relationship"};
445 }
446
447 // Add to stored relationships
448 storedRelationships.insert({line, item});
449
450 // Add to scene
451 line->initialize(firstClassNode, secondClassNode, firstClassNode == secondClassNode);
452 addItem(line);
453 firstClassNode->addLine(line);
454 secondClassNode->addLine(line);
455 }
456}

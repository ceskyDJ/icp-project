42{

62 QString::fromStdString(":" + object->getInstanceClass().getReferredClassName() + "\n" + object->getName()));
63 if(!object->getInstanceClass().isValid())
64 painter->setPen(penToUse);
65

74 painter->drawRect(lifeBox);
75

77 painter->setPen(lifeLineBorderPen);
78 painter->drawLine(0, headerHeight + lifeSpaceStart, textWidth, headerHeight + lifeSpaceStart);
79 painter->drawLine(0, height, textWidth, height);
80

87 lifeBox.y() + lifeBox.height() + destroyCrossSize * 0.5);

89 crossRect.x() + crossRect.width(), crossRect.y() + crossRect.height());

136void ObjectGraphicsItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
137{
138 if(this->cursor() == Qt::ClosedHandCursor) // lifebox is going to be moved
139 {
140 QRectF lifebox = lifeBoxRect();
141 qreal dy = event->pos().y() - pressedPos.y();
142
143 //if new position of lifebox is in life area, then moves self by coordinates
144 if((lifebox.y() + dy >= headerHeight + lifeSpaceStart) && (lifebox.y() + dy + lifebox.height() <= height))
145 {
146 qreal maxBoxHeight = height - lifeSpaceStart - headerHeight;
147 object->setLifeStart((lifebox.y() + dy - lifeSpaceStart - headerHeight) / maxBoxHeight);
148 moveAllMessages(dy);
149 }
150
151 }
152 else if(this->cursor() == Qt::SizeVerCursor)//resizing mode
153 {
154 QRectF lifebox = lifeBoxRect();
155 qreal dy = event->pos().y() - pressedPos.y();
156 qreal newLength = (lifebox.height() + dy) / (height - lifeSpaceStart - headerHeight);
157 if (lifebox.y() + dy + lifebox.height() > height)//lifebox could be outside of borders
158 {
159 pressedPos = event->pos();
160 return;
161 }
162 else if(newLength <= 0) // length is too small
163 return;
164 if(destroyed) //destroyed flag on
165 getDestroyMessage()->moveLine(dy, true);
166 else //everything ok
167 object->setLifeLength(newLength);
168 }
169 else if (this->cursor() == Qt::SizeAllCursor) // Possibility of moving object in scene
170 {
171 int dx = (int)((event->scenePos().x() - this->scenePos().x()) / (objectPadding - moveTollerance));
172 if(dx == -1)
173 emitter.emitMoveLeft();
174 else if (dx == 1)
175 emitter.emitMoveRight();
176 }
177 pressedPos = event->pos();
178}

246void ObjectGraphicsItem::showEditDialog(bool logChange)
247{
248 ObjectGraphicsItemEditDialog dialog{QString::fromStdString(object->getName()),
249 QString::fromStdString(object->getInstanceClass().getReferredClassName()), classDiagram};
250 int result = dialog.exec();

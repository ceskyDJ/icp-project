45void ObjectGraphicsItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *)
46{
47 prepareGeometryChange();
48 painter->fillRect(boundingRect(), Qt::white);
49 QPen penToUse;
50 if(option->state & QStyle::State_Selected)
51 penToUse = selectedPen;
52 else
53 {
54 penToUse = regularPen;
55 penToUse.setColor(drawColor);
56 }
57
58 painter->setPen(penToUse);
59 qreal textWidth = width();
60
61 //drawing header
62 painter->drawRect(0,0, textWidth, headerHeight);
63 if(!object->getInstanceClass().isValid())
64 painter->setPen(unknownObject);
65 painter->drawText(0,0, textWidth, headerHeight, Qt::AlignCenter,
66 QString::fromStdString(":" + object->getInstanceClass().getReferredClassName() + "\n" + object->getName()));
67 if(!object->getInstanceClass().isValid())
68 painter->setPen(penToUse);
69
70 //drawing life line
71 painter->setPen(lifeLinePen);
72 painter->drawLine(textWidth / 2, headerHeight, textWidth / 2, height);
73 painter->setPen(penToUse);
74
75 //drawing lifebox
76 QRectF lifeBox = lifeBoxRect();
77 painter->fillRect(lifeBox, Qt::white);
78 painter->drawRect(lifeBox);
79
80 //drawing end and start of life area
81 painter->setPen(lifeLineBorderPen);
82 painter->drawLine(0, headerHeight + lifeSpaceStart, textWidth, headerHeight + lifeSpaceStart);
83 painter->drawLine(0, height, textWidth, height);
84
85 if(destroyed)//draw destroy cross
86 {
87 painter->setPen(QPen{Qt::black, 2, Qt::SolidLine});
88 QRectF crossRect{-destroyCrossSize * 0.5, -destroyCrossSize * 0.5,
89 destroyCrossSize, destroyCrossSize};
90 crossRect.translate(lifeBox.x() + lifeboxWidth * 0.5,
91 lifeBox.y() + lifeBox.height() + destroyCrossSize * 0.5);
92 painter->drawLine(crossRect.x(), crossRect.y(),
93 crossRect.x() + crossRect.width(), crossRect.y() + crossRect.height());
94 painter->drawLine(crossRect.x() + crossRect.width(), crossRect.y(),
95 crossRect.x(), crossRect.y() + crossRect.height());
96 }
97}

152void ObjectGraphicsItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
153{
154 if(this->cursor() == Qt::ClosedHandCursor) // lifebox is going to be moved
155 {
156 QRectF lifebox = lifeBoxRect();
157 qreal dy = event->pos().y() - pressedPos.y();
158
159 //if new position of lifebox is in life area, then moves self by coordinates
160 if((lifebox.y() + dy >= headerHeight + lifeSpaceStart) && (lifebox.y() + dy + lifebox.height() <= height))
161 {
162 qreal maxBoxHeight = height - lifeSpaceStart - headerHeight;
163 object->setLifeStart((lifebox.y() + dy - lifeSpaceStart - headerHeight) / maxBoxHeight);
164 moveAllMessages(dy);
165 }
166
167 }
168 else if(this->cursor() == Qt::SizeVerCursor)//resizing mode
169 {
170 QRectF lifebox = lifeBoxRect();
171 qreal dy = event->pos().y() - pressedPos.y();
172 qreal newLength = (lifebox.height() + dy) / (height - lifeSpaceStart - headerHeight);
173 if (lifebox.y() + dy + lifebox.height() > height)//lifebox could be outside of borders
174 {
175 pressedPos = event->pos();
176 return;
177 }
178 else if (newLength <= 0) // length is too small
179 return;
180 if (destroyed) //destroyed flag on
181 getDestroyMessage()->moveLine(dy, true);
182 else //everything ok
183 object->setLifeLength(newLength);
184 }
185 else if (this->cursor() == Qt::SizeAllCursor) // Possibility of moving object in scene
186 {
187 int dx = (int)((event->scenePos().x() - this->scenePos().x()) / (objectPadding - moveTollerance));
188 if(dx == -1)
189 emitter.emitMoveLeft();
190 else if (dx == 1)
191 emitter.emitMoveRight();
192 }
193 pressedPos = event->pos();
194}

262void ObjectGraphicsItem::showEditDialog(bool logChange)
263{
264 ObjectGraphicsItemEditDialog dialog{QString::fromStdString(object->getName()),
265 QString::fromStdString(object->getInstanceClass().getReferredClassName()), classDiagram};
266 int result = dialog.exec();
267 if (result == QDialog::Accepted) {
268 object->setName(dialog.getObjectName().toStdString());
269 object->setInstanceClass(dialog.getClassRef());
270 //update class references in all messages sent to this item.
271 updateMessagesClassReference(dialog.getClassRef());
272 update();
273
274 if (logChange) {
275 sceneUpdateObservable->sceneChanged();
276 }
277 } else if (result == ObjectGraphicsItemEditDialog::remove) {
278 emitter.emitRemoveObjectSignal(logChange);
279 }
280}

39void ObjectGraphicsItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *)
40{
41 prepareGeometryChange();
42 painter->fillRect(boundingRect(), Qt::white);
43 QPen penToUse;
44 if(option->state & QStyle::State_Selected)
45 penToUse = selectedPen;
46 else
47 {
48 penToUse = regularPen;
49 penToUse.setColor(drawColor);
50 }
51
52 painter->setPen(penToUse);
53 qreal textWidth = width();
54
55 //drawing header
56 painter->drawRect(0,0, textWidth, headerHeight);
57 if(!object->getInstanceClass().isValid())
58 painter->setPen(unknownObject);
59 painter->drawText(0,0, textWidth, headerHeight, Qt::AlignCenter,
60 QString::fromStdString(":" + object->getInstanceClass().getReferredClassName() + "\n" + object->getName()));
61 if(!object->getInstanceClass().isValid())
62 painter->setPen(penToUse);
63
64 //drawing life line
65 painter->setPen(lifeLinePen);
66 painter->drawLine(textWidth / 2, headerHeight, textWidth / 2, height);
67 painter->setPen(penToUse);
68
69 //drawing lifebox
70 QRectF lifeBox = lifeBoxRect();
71 painter->fillRect(lifeBox, Qt::white);
72 painter->drawRect(lifeBox);
73
74 //drawing end and start of life area
75 painter->setPen(lifeLineBorderPen);
76 painter->drawLine(0, headerHeight + lifeSpaceStart, textWidth, headerHeight + lifeSpaceStart);
77 painter->drawLine(0, height, textWidth, height);
78
79 if(destroyed)//draw destroy cross
80 {
81 painter->setPen(QPen{Qt::black, 2, Qt::SolidLine});
82 QRectF crossRect{-destroyCrossSize * 0.5, -destroyCrossSize * 0.5,
83 destroyCrossSize, destroyCrossSize};
84 crossRect.translate(lifeBox.x() + lifeboxWidth * 0.5,
85 lifeBox.y() + lifeBox.height() + destroyCrossSize * 0.5);
86 painter->drawLine(crossRect.x(), crossRect.y(),
87 crossRect.x() + crossRect.width(), crossRect.y() + crossRect.height());
88 painter->drawLine(crossRect.x() + crossRect.width(), crossRect.y(),
89 crossRect.x(), crossRect.y() + crossRect.height());
90 }
91}

134void ObjectGraphicsItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
135{
136 setSelected(true);
137 if(this->cursor() == Qt::ClosedHandCursor)
138 {
139 QRectF lifebox = lifeBoxRect();
140 qreal dy = event->pos().y() - pressedPos.y();
141
142 //if new position of lifebox is in life area, then moves self by coordinates
143 if((lifebox.y() + dy >= headerHeight + lifeSpaceStart) && (lifebox.y() + dy + lifebox.height() <= height))
144 {
145 qreal maxBoxHeight = height - lifeSpaceStart - headerHeight;
146 object->setLifeStart((lifebox.y() + dy - lifeSpaceStart - headerHeight) / maxBoxHeight);
147 moveAllMessages(dy);
148 }
149
150 }
151 else if(this->cursor() == Qt::SizeVerCursor)//resizing mode
152 {
153 QRectF lifebox = lifeBoxRect();
154 qreal dy = event->pos().y() - pressedPos.y();
155 qreal newLength = (lifebox.height() + dy) / (height - lifeSpaceStart - headerHeight);
156 if (lifebox.y() + dy + lifebox.height() > height)//lifebox could be outside of borders
157 {
158 pressedPos = event->pos();
159 return;
160 }
161 else if(newLength <= 0) // length is too small
162 return;
163 if(destroyed) //destroyed flag on
164 getDestroyMessage()->moveLine(dy, true);
165 else //everything ok
166 object->setLifeLength(newLength);
167 }
168 pressedPos = event->pos();
169}

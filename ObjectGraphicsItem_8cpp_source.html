127void ObjectGraphicsItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
128{
129 setSelected(true);
130 if(this->cursor() == Qt::ClosedHandCursor)
131 {
132 QRectF lifebox = lifeBoxRect();
133 qreal dy = event->pos().y() - pressedPos.y();
134
135 //if new position of lifebox is in life area, then moves self by coordinates
136 if((lifebox.y() + dy >= headerHeight + lifeSpaceStart) && (lifebox.y() + dy + lifebox.height() <= height))
137 {
138 qreal maxBoxHeight = height - lifeSpaceStart - headerHeight;
139 object->setLifeStart((lifebox.y() + dy - lifeSpaceStart - headerHeight) / maxBoxHeight);
140 moveAllMessages(dy);
141 }
142
143 }
144 else if(this->cursor() == Qt::SizeVerCursor)//resizing mode
145 {
146 QRectF lifebox = lifeBoxRect();
147 qreal dy = event->pos().y() - pressedPos.y();
148 qreal newLength = (lifebox.height() + dy) / (height - lifeSpaceStart - headerHeight);
149 if (lifebox.y() + dy + lifebox.height() > height)//lifebox could be outside of borders
150 {
151 pressedPos = event->pos();
152 return;
153 }
154 else if(newLength <= 0) // length is too small
155 return;
156 if(destroyed) //destroyed flag on
157 getDestroyMessage()->moveLine(dy, true);
158 else //everything ok
159 object->setLifeLength(newLength);
160 }
161 pressedPos = event->pos();
162}

234void ClassEditDialog::onConfirmChangesPressed()
235{
236 //get class attributes
237 std::vector<ClassAttribute> newAttribs;
238 for (int i = 0; i < attributeLayout->count(); ++i)
239 {
240 QWidget *widget = attributeLayout->itemAt(i)->widget();
241 AttributeEditWidget *attrWidget = dynamic_cast<AttributeEditWidget *>(widget);
242 newAttribs.push_back(*(attrWidget->getAttribute()));
243 }
244
245 //get class methods
246 std::vector<ClassMethod> newMethods;
247 for (int i = 0; i < methodLayout->count(); ++i)
248 {
249 QWidget *widget = methodLayout->itemAt(i)->widget();
250 MethodEditWidget *methodParWidget = dynamic_cast<MethodEditWidget *>(widget);
251 newMethods.push_back(*(methodParWidget->getMethod()));
252 }
253
254 classEntity->setAttributes(newAttribs);
255 classEntity->setMethods(newMethods);
256 accept();
257}

235void ClassEditDialog::onConfirmChangesPressed()
236{
237 //get class attributes
238 std::vector<ClassAttribute> newAttribs;
239 for (int i = 0; i < attributeLayout->count(); ++i)
240 {
241 QWidget *widget = attributeLayout->itemAt(i)->widget();
242 AttributeEditWidget *attrWidget = dynamic_cast<AttributeEditWidget *>(widget);
243 newAttribs.push_back(*(attrWidget->getAttribute()));
244 }
245
246 //get class methods
247 std::vector<ClassMethod> newMethods;
248 for (int i = 0; i < methodLayout->count(); ++i)
249 {
250 QWidget *widget = methodLayout->itemAt(i)->widget();
251 MethodEditWidget *methodParWidget = dynamic_cast<MethodEditWidget *>(widget);
252 newMethods.push_back(*(methodParWidget->getMethod()));
253 }
254
255 classEntity.setAttributes(newAttribs);
256 classEntity.setMethods(newMethods);
257 accept();
258}

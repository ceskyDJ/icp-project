236void ClassEditDialog::onConfirmChangesPressed()
237{
238 //get class attributes
239 std::vector<ClassAttribute> newAttribs;
240 for (int i = 0; i < attributeLayout->count(); ++i)
241 {
242 QWidget *widget = attributeLayout->itemAt(i)->widget();
243 AttributeEditWidget *attrWidget = dynamic_cast<AttributeEditWidget *>(widget);
244 newAttribs.push_back(*(attrWidget->getAttribute()));
245 }
246
247 //get class methods
248 std::vector<ClassMethod> newMethods;
249 for (int i = 0; i < methodLayout->count(); ++i)
250 {
251 QWidget *widget = methodLayout->itemAt(i)->widget();
252 MethodEditWidget *methodParWidget = dynamic_cast<MethodEditWidget *>(widget);
253 newMethods.push_back(*(methodParWidget->getMethod()));
254 }
255
256 classEntity.setAttributes(newAttribs);
257 classEntity.setMethods(newMethods);
258 accept();
259}

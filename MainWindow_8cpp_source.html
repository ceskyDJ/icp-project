269std::tuple<CustomScene *, TabWidget *> MainWindow::createScene(
270 const SceneType &type,
271 const QString &name,
272 bool closeable
273) {
274 // Numbering for unique names
275 static unsigned int classDiagramNumber = 1;
276 static unsigned int sequenceDiagramNumber = 1;
277
278 // Create new scene instance
279 CustomScene *newScene{};
280 QString defaultName{};
281 switch (type) {
282 case SceneType::ClassDiagram:
283 newScene = new ClassDiagramScene{this, classDiagramManager, sceneUpdateObservable};
284 defaultName = "New class diagram " + QString::number(classDiagramNumber++);
285 break;
286 case SceneType::SequenceDiagram:
287 ClassDiagram *classDiagram = getClassDiagramScene()->getClassDiagram();
288
289 newScene = new SequenceDiagramScene{this, sequenceDiagramManager, sceneUpdateObservable, classDiagram};
290 defaultName = "New sequence diagram " + QString::number(sequenceDiagramNumber++);
291 break;
292 }
293
294 scenes.push_back(newScene);
295
296 // Create tab for scene
297 auto newTab = new TabWidget{(name.isEmpty() ? defaultName : name), newScene, closeable};
298 tabs.push_back(newTab);
299
300 QAction *action = tabBar->addWidget(newTab);
301 tabBarActions.insert({newTab, action});
302
303 // Connect signals from tab buttons
304 connect(newTab->getTabButton(), &QPushButton::pressed, this, &MainWindow::selectTab);
305 connect(newTab->getCloseButton(), &QPushButton::pressed, this, &MainWindow::closeTab);
306
307 return {newScene, newTab};
308}

622void MainWindow::openButtonClicked()
623{
624 QString file = QFileDialog::getOpenFileName(
625 this,
626 "Choose diagram to open",
627 "",
628 "Class diagrams (*.cls.xml);;Sequence diagrams (*.seq.xml)",
629 nullptr,
630 QFileDialog::HideNameFilterDetails
631 );
632
633 // Get short file name (end of path)
634 QFileInfo fileInfo{file};
635 QString shortFileName{fileInfo.fileName()};
636
637 if (file.endsWith(".cls.xml")) {
638 // Class diagram
639 ClassDiagramScene *classDiagramScene = getClassDiagramScene();
640
641 if (!classDiagramScene->items().isEmpty() && !classDiagramScene->isSaved()) {
642 auto clickedButton = QMessageBox::question(
643 this,
644 "Unsaved class diagram",
645 "You are trying to open new class diagram before saving the edited one. Are you sure to"
646 " continue? Canvas content will be replaced with loaded diagram."
647 );
648
649 if (clickedButton == QMessageBox::No) {
650 // Stop here, nothing will be loaded
651 return;
652 }
653 }
654
655 // Switch to class diagram scene
656 // Class diagram is always in the first tab
657 tabs[0]->setLabel(shortFileName);
658 setActiveTab(tabs[0]);
659 } else if (file.endsWith(".seq.xml")) {
660 // Sequence diagram
661
662 // One file can be used only in one scene
663 if (isFileUsedBySomeScene(file)) {
664 QMessageBox::critical(
665 this,
666 "Diagram opening error",
667 "Selected file is already used by some of diagrams you are editing. Select different file or close"
668 " edited diagram from the file before opening again."
669 );
670
671 // Stop here, nothing will be loaded
672 return;
673 }
674
675 // Add new scene for diagram to be loaded and switch to it
676 auto sceneData = createScene(SceneType::SequenceDiagram, shortFileName);
677 setActiveTab(std::get<1>(sceneData));
678 } else {
679 // File with bad extension --> stop processing
680 QMessageBox::critical(this, "Diagram opening error", "Selected file has bad extension.");
681
682 return;
683 }
684
685 // Remember source file
686 currentScene->setTargetFile(file);
687
688 // Load diagram from file
689 currentScene->loadFromFile();
690
691 // Update tab
692 updateTab(currentTab);
693}

721void MainWindow::saveAsButtonClicked() // NOLINT(misc-no-recursion)
722{
723 QString currentTargetFile{currentScene->getTargetFile()};
724
725 // Choose filter by diagram type
726 QString filter{};
727 QString defaultFileName{};
728 if (typeid(*currentScene) == typeid(ClassDiagramScene)) {
729 // Class diagram
730 filter = "Class diagrams (*.cls.xml)";
731 defaultFileName = "new-class-diagram.cls.xml";
732 } else {
733 // Sequence diagram
734 filter = "Sequence diagrams (*.seq.xml)";
735 defaultFileName = "new-sequence-diagram.seq.xml";
736 }
737
738 // Prompt user to select the target file
739 QString file = QFileDialog::getSaveFileName(
740 this,
741 "Choose target file for saving diagram",
742 currentTargetFile.isEmpty() ? defaultFileName : currentTargetFile,
743 filter,
744 nullptr,
745 QFileDialog::HideNameFilterDetails
746 );
747
748 if (file.isEmpty()) {
749 // Dialog has been canceled
750 return;
751 }
752
753 // Check for extension validity
754 if (typeid(*currentScene) == typeid(ClassDiagramScene) && !file.endsWith(".cls.xml")) {
755 // Class diagram
756 QMessageBox::critical(
757 this,
758 "Class diagram saving error",
759 "Your file has invalid extension. Class diagrams must be saved with name like: \"*.cls.xml\" (without"
760 " quotes)."
761 );
762
763 // Stop here, nothing will be saved
764 return;
765 } else if(typeid(*currentScene) == typeid(SequenceDiagramScene) && !file.endsWith(".seq.xml")) {
766 // Sequence diagram
767 QMessageBox::critical(
768 this,
769 "Sequence diagram saving error",
770 "Your file has invalid extension. Sequence diagrams must be saved with name like: \"*.seq.xml\""
771 " (without quotes)."
772 );
773
774 // Stop here, nothing will be saved
775 return;
776 }
777
778 // File used by some scene cannot be rewritten
779 if (isFileUsedBySomeScene(file) && currentTargetFile != file) {
780 QMessageBox::critical(
781 this,
782 "Diagram saving error",
783 "Selected file is used as a target for saving one of edited diagrams. Choose different file for saving"
784 " the diagram, please."
785 );
786
787 // Stop here, nothing will be saved
788 return;
789 }
790
791 // Update target file
792 currentScene->setTargetFile(file);
793
794 // Get short file name (end of path)
795 QFileInfo fileInfo{file};
796 QString shortFileName{fileInfo.fileName()};
797
798 // Update tab label
799 currentTab->setLabel(shortFileName);
800
801 // Now just save diagram into selected file
802 saveButtonClicked();
803}

836void MainWindow::closeTab()
837{
838 auto button = qobject_cast<QPushButton *>(sender());
839 auto tab = qobject_cast<TabWidget *>(button->parentWidget());
840 CustomScene *scene = tab->getScene();
841
842 // Check for closing unsaved scene
843 if (!scene->items().empty() && !scene->isSaved()) {
844 auto clickedButton = QMessageBox::question(
845 this,
846 "Unsaved class diagram",
847 "You are trying to close diagram before saving it. Are you sure to continue? All your unsaved"
848 " changes will be lost."
849 );
850
851 if (clickedButton == QMessageBox::No) {
852 // Stop here, tab won't be closed
853 return;
854 }
855 }
856
857 // Switch to class diagram tab when active tab will be closed
858 if (tab == currentTab) {
859 setActiveTab(tabs[0]);
860 }
861
862 // Remove tab from tab bar
863 tabBar->removeAction(tabBarActions.find(tab)->second);
864 tabBarActions.erase(tab);
865
866 // Remove tab from the list and memory
867 for (auto iterator = tabs.begin(); iterator < tabs.end(); iterator++) {
868 if (*iterator == tab) {
869 tabs.erase(iterator);
870
871 delete tab;
872 break;
873 }
874 }
875
876 // Remove scene from existing scenes
877 for (auto iterator = scenes.begin(); iterator < scenes.end(); iterator++) {
878 if (*iterator == scene) {
879 scenes.erase(iterator);
880
881 delete scene;
882 return;
883 }
884 }

273std::tuple<CustomScene *, TabWidget *> MainWindow::createScene(
274 const SceneType &type,
275 const QString &name,
276 bool closeable
277) {
278 // Numbering for unique names
279 static unsigned int classDiagramNumber = 1;
280 static unsigned int sequenceDiagramNumber = 1;
281
282 // Create new scene instance
283 CustomScene *newScene{};
284 QString defaultName{};
285 switch (type) {
286 case SceneType::ClassDiagram:
287 newScene = new ClassDiagramScene{this, classDiagramManager, sceneUpdateObservable};
288 defaultName = "New class diagram " + QString::number(classDiagramNumber++);
289 break;
290 case SceneType::SequenceDiagram:
291 ClassDiagram *classDiagram = getClassDiagramScene()->getClassDiagram();
292
293 newScene = new SequenceDiagramScene{this, sequenceDiagramManager, sceneUpdateObservable, classDiagram};
294 defaultName = "New sequence diagram " + QString::number(sequenceDiagramNumber++);
295 break;
296 }
297
298 scenes.push_back(newScene);
299
300 // Create tab for scene
301 auto newTab = new TabWidget{(name.isEmpty() ? defaultName : name), newScene, closeable};
302 tabs.push_back(newTab);
303
304 QAction *action = tabBar->addWidget(newTab);
305 tabBarActions.insert({newTab, action});
306
307 // Connect signals from tab buttons
308 connect(newTab->getTabButton(), &QPushButton::pressed, this, &MainWindow::selectTab);
309 connect(newTab->getCloseButton(), &QPushButton::pressed, this, &MainWindow::closeTab);
310
311 return {newScene, newTab};
312}

625void MainWindow::openButtonClicked()
626{
627 QString file = QFileDialog::getOpenFileName(
628 this,
629 "Choose diagram to open",
630 "",
631 "Class diagrams (*.cls.xml);;Sequence diagrams (*.seq.xml)",
632 nullptr,
633 QFileDialog::HideNameFilterDetails
634 );
635
636 // Get short file name (end of path)
637 QFileInfo fileInfo{file};
638 QString shortFileName{fileInfo.fileName()};
639
640 if (file.endsWith(".cls.xml")) {
641 // Class diagram
642 ClassDiagramScene *classDiagramScene = getClassDiagramScene();
643
644 if (!classDiagramScene->items().isEmpty() && !classDiagramScene->isSaved()) {
645 auto clickedButton = QMessageBox::question(
646 this,
647 "Unsaved class diagram",
648 "You are trying to open new class diagram before saving the edited one. Are you sure to"
649 " continue? Canvas content will be replaced with loaded diagram."
650 );
651
652 if (clickedButton == QMessageBox::No) {
653 // Stop here, nothing will be loaded
654 return;
655 }
656 }
657
658 // Switch to class diagram scene
659 // Class diagram is always in the first tab
660 tabs[0]->setLabel(shortFileName);
661 setActiveTab(tabs[0]);
662 } else if (file.endsWith(".seq.xml")) {
663 // Sequence diagram
664
665 // One file can be used only in one scene
666 if (isFileUsedBySomeScene(file)) {
667 QMessageBox::critical(
668 this,
669 "Diagram opening error",
670 "Selected file is already used by some of diagrams you are editing. Select different file or close"
671 " edited diagram from the file before opening again."
672 );
673
674 // Stop here, nothing will be loaded
675 return;
676 }
677
678 // Add new scene for diagram to be loaded and switch to it
679 auto sceneData = createScene(SceneType::SequenceDiagram, shortFileName);
680 setActiveTab(std::get<1>(sceneData));
681 } else {
682 // File with bad extension --> stop processing
683 QMessageBox::critical(this, "Diagram opening error", "Selected file has bad extension.");
684
685 return;
686 }
687
688 // Remember source file
689 currentScene->setTargetFile(file);
690
691 // Load diagram from file
692 currentScene->loadFromFile();
693
694 // Update tab
695 updateTab(currentTab);
696}

724void MainWindow::saveAsButtonClicked() // NOLINT(misc-no-recursion)
725{
726 QString currentTargetFile{currentScene->getTargetFile()};
727
728 // Choose filter by diagram type
729 QString filter{};
730 QString defaultFileName{};
731 if (typeid(*currentScene) == typeid(ClassDiagramScene)) {
732 // Class diagram
733 filter = "Class diagrams (*.cls.xml)";
734 defaultFileName = "new-class-diagram.cls.xml";
735 } else {
736 // Sequence diagram
737 filter = "Sequence diagrams (*.seq.xml)";
738 defaultFileName = "new-sequence-diagram.seq.xml";
739 }
740
741 // Prompt user to select the target file
742 QString file = QFileDialog::getSaveFileName(
743 this,
744 "Choose target file for saving diagram",
745 currentTargetFile.isEmpty() ? defaultFileName : currentTargetFile,
746 filter,
747 nullptr,
748 QFileDialog::HideNameFilterDetails
749 );
750
751 if (file.isEmpty()) {
752 // Dialog has been canceled
753 return;
754 }
755
756 // Check for extension validity
757 if (typeid(*currentScene) == typeid(ClassDiagramScene) && !file.endsWith(".cls.xml")) {
758 // Class diagram
759 QMessageBox::critical(
760 this,
761 "Class diagram saving error",
762 "Your file has invalid extension. Class diagrams must be saved with name like: \"*.cls.xml\" (without"
763 " quotes)."
764 );
765
766 // Stop here, nothing will be saved
767 return;
768 } else if(typeid(*currentScene) == typeid(SequenceDiagramScene) && !file.endsWith(".seq.xml")) {
769 // Sequence diagram
770 QMessageBox::critical(
771 this,
772 "Sequence diagram saving error",
773 "Your file has invalid extension. Sequence diagrams must be saved with name like: \"*.seq.xml\""
774 " (without quotes)."
775 );
776
777 // Stop here, nothing will be saved
778 return;
779 }
780
781 // File used by some scene cannot be rewritten
782 if (isFileUsedBySomeScene(file) && currentTargetFile != file) {
783 QMessageBox::critical(
784 this,
785 "Diagram saving error",
786 "Selected file is used as a target for saving one of edited diagrams. Choose different file for saving"
787 " the diagram, please."
788 );
789
790 // Stop here, nothing will be saved
791 return;
792 }
793
794 // Update target file
795 currentScene->setTargetFile(file);
796
797 // Get short file name (end of path)
798 QFileInfo fileInfo{file};
799 QString shortFileName{fileInfo.fileName()};
800
801 // Update tab label
802 currentTab->setLabel(shortFileName);
803
804 // Now just save diagram into selected file
805 saveButtonClicked();
806}

839void MainWindow::closeTab()
840{
841 auto button = qobject_cast<QPushButton *>(sender());
842 auto tab = qobject_cast<TabWidget *>(button->parentWidget());
843 CustomScene *scene = tab->getScene();
844
845 // Check for closing unsaved scene
846 if (!scene->items().empty() && !scene->isSaved()) {
847 auto clickedButton = QMessageBox::question(
848 this,
849 "Unsaved class diagram",
850 "You are trying to close diagram before saving it. Are you sure to continue? All your unsaved"
851 " changes will be lost."
852 );
853
854 if (clickedButton == QMessageBox::No) {
855 // Stop here, tab won't be closed
856 return;
857 }
858 }
859
860 // Switch to class diagram tab when active tab will be closed
861 if (tab == currentTab) {
862 setActiveTab(tabs[0]);
863 }
864
865 // Remove tab from tab bar
866 tabBar->removeAction(tabBarActions.find(tab)->second);
867 tabBarActions.erase(tab);
868
869 // Remove tab from the list and memory
870 for (auto iterator = tabs.begin(); iterator < tabs.end(); iterator++) {
871 if (*iterator == tab) {
872 tabs.erase(iterator);
873
874 delete tab;
875 break;
876 }
877 }
878
879 // Remove scene from existing scenes
880 for (auto iterator = scenes.begin(); iterator < scenes.end(); iterator++) {
881 if (*iterator == scene) {
882 scenes.erase(iterator);
883
884 delete scene;
885 return;
886 }
887 }

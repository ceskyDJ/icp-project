200std::tuple<CustomScene *, TabWidget *> MainWindow::createScene(
201 const SceneType &type,
202 const QString &name,
203 bool closeable
204) {
205 // Numbering for unique names
206 static unsigned int classDiagramNumber = 1;
207 static unsigned int sequenceDiagramNumber = 1;
208
209 // Create new scene instance
210 CustomScene *newScene{};
211 QString defaultName{};
212 switch (type) {
213 case SceneType::ClassDiagram:
214 newScene = new ClassDiagramScene{this, classDiagramManager, sceneUpdateObservable};
215 defaultName = "New class diagram " + QString::number(classDiagramNumber++);
216 break;
217 case SceneType::SequenceDiagram:
218 ClassDiagram *classDiagram = getClassDiagramScene()->getClassDiagram();
219
220 newScene = new SequenceDiagramScene{this, sequenceDiagramManager, sceneUpdateObservable, classDiagram};
221 defaultName = "New sequence diagram " + QString::number(sequenceDiagramNumber++);
222 break;
223 }
224
225 scenes.push_back(newScene);
226
227 // Create tab for scene
228 auto newTab = new TabWidget{(name.isEmpty() ? defaultName : name), newScene, closeable};
229 tabs.push_back(newTab);
230
231 QAction *action = tabBar->addWidget(newTab);
232 tabBarActions.insert({newTab, action});
233
234 // Connect signals from tab buttons
235 connect(newTab->getTabButton(), &QPushButton::pressed, this, &MainWindow::selectTab);
236 connect(newTab->getCloseButton(), &QPushButton::pressed, this, &MainWindow::closeTab);
237
238 return {newScene, newTab};
239}

443void MainWindow::openButtonClicked()
444{
445 QString file = QFileDialog::getOpenFileName(
446 this,
447 "Choose diagram to open",
448 "",
449 "Class diagrams (*.cls.xml);;Sequence diagrams (*.seq.xml)",
450 nullptr,
451 QFileDialog::HideNameFilterDetails
452 );
453
454 // Get short file name (end of path)
455 QFileInfo fileInfo{file};
456 QString shortFileName{fileInfo.fileName()};
457
458 if (file.endsWith(".cls.xml")) {
459 // Class diagram
460 ClassDiagramScene *classDiagramScene = getClassDiagramScene();
461
462 if (!classDiagramScene->items().isEmpty() && !classDiagramScene->isSaved()) {
463 auto clickedButton = QMessageBox::question(
464 this,
465 "Unsaved class diagram",
466 "You are trying to open new class diagram before saving the edited one. Are you sure to"
467 " continue? Canvas content will be replaced with loaded diagram."
468 );
469
470 if (clickedButton == QMessageBox::No) {
471 // Stop here, nothing will be loaded
472 return;
473 }
474 }
475
476 // Switch to class diagram scene
477 // Class diagram is always in the first tab
478 tabs[0]->setLabel(shortFileName);
479 setActiveTab(tabs[0]);
480 } else if (file.endsWith(".seq.xml")) {
481 // Sequence diagram
482
483 // One file can be used only in one scene
484 if (isFileUsedBySomeScene(file)) {
485 QMessageBox::critical(
486 this,
487 "Diagram opening error",
488 "Selected file is already used by some of diagrams you are editing. Select different file or close"
489 " edited diagram from the file before opening again."
490 );
491
492 // Stop here, nothing will be loaded
493 return;
494 }
495
496 // Add new scene for diagram to be loaded and switch to it
497 auto sceneData = createScene(SceneType::SequenceDiagram, shortFileName);
498 setActiveTab(std::get<1>(sceneData));
499 } else {
500 // File with bad extension --> stop processing
501 QMessageBox::critical(this, "Diagram opening error", "Selected file has bad extension.");
502
503 return;
504 }
505
506 // Remember source file
507 currentScene->setTargetFile(file);
508
509 // Load diagram from file
510 currentScene->loadFromFile();
511
512 // Update tab
513 updateTab(currentTab);
514}

542void MainWindow::saveAsButtonClicked() // NOLINT(misc-no-recursion)
543{
544 QString currentTargetFile{currentScene->getTargetFile()};
545
546 // Choose filter by diagram type
547 QString filter{};
548 QString defaultFileName{};
549 if (typeid(*currentScene) == typeid(ClassDiagramScene)) {
550 // Class diagram
551 filter = "Class diagrams (*.cls.xml)";
552 defaultFileName = "new-class-diagram.cls.xml";
553 } else {
554 // Sequence diagram
555 filter = "Sequence diagrams (*.seq.xml)";
556 defaultFileName = "new-sequence-diagram.seq.xml";
557 }
558
559 // Prompt user to select the target file
560 QString file = QFileDialog::getSaveFileName(
561 this,
562 "Choose target file for saving diagram",
563 currentTargetFile.isEmpty() ? defaultFileName : currentTargetFile,
564 filter,
565 nullptr,
566 QFileDialog::HideNameFilterDetails
567 );
568
569 if (file.isEmpty()) {
570 // Dialog has been canceled
571 return;
572 }
573
574 // Check for extension validity
575 if (typeid(*currentScene) == typeid(ClassDiagramScene) && !file.endsWith(".cls.xml")) {
576 // Class diagram
577 QMessageBox::critical(
578 this,
579 "Class diagram saving error",
580 "Your file has invalid extension. Class diagrams must be saved with name like: \"*.cls.xml\" (without"
581 " quotes)."
582 );
583
584 // Stop here, nothing will be saved
585 return;
586 } else if(typeid(*currentScene) == typeid(SequenceDiagramScene) && !file.endsWith(".seq.xml")) {
587 // Sequence diagram
588 QMessageBox::critical(
589 this,
590 "Sequence diagram saving error",
591 "Your file has invalid extension. Sequence diagrams must be saved with name like: \"*.seq.xml\""
592 " (without quotes)."
593 );
594
595 // Stop here, nothing will be saved
596 return;
597 }
598
599 // File used by some scene cannot be rewritten
600 if (isFileUsedBySomeScene(file) && currentTargetFile != file) {
601 QMessageBox::critical(
602 this,
603 "Diagram saving error",
604 "Selected file is used as a target for saving one of edited diagrams. Choose different file for saving"
605 " the diagram, please."
606 );
607
608 // Stop here, nothing will be saved
609 return;
610 }
611
612 // Update target file
613 currentScene->setTargetFile(file);
614
615 // Get short file name (end of path)
616 QFileInfo fileInfo{file};
617 QString shortFileName{fileInfo.fileName()};
618
619 // Update tab label
620 currentTab->setLabel(shortFileName);
621
622 // Now just save diagram into selected file
623 saveButtonClicked();
624}

657void MainWindow::closeTab()
658{
659 auto button = qobject_cast<QPushButton *>(sender());
660 auto tab = qobject_cast<TabWidget *>(button->parentWidget());
661 CustomScene *scene = tab->getScene();
662
663 // Check for closing unsaved scene
664 if (!scene->items().empty() && !scene->isSaved()) {
665 auto clickedButton = QMessageBox::question(
666 this,
667 "Unsaved class diagram",
668 "You are trying to close diagram before saving it. Are you sure to continue? All your unsaved"
669 " changes will be lost."
670 );
671
672 if (clickedButton == QMessageBox::No) {
673 // Stop here, tab won't be closed
674 return;
675 }
676 }
677
678 // Switch to class diagram tab when active tab will be closed
679 if (tab == currentTab) {
680 setActiveTab(tabs[0]);
681 }
682
683 // Remove tab from tab bar
684 tabBar->removeAction(tabBarActions.find(tab)->second);
685 tabBarActions.erase(tab);
686
687 // Remove tab from the list and memory
688 for (auto iterator = tabs.begin(); iterator < tabs.end(); iterator++) {
689 if (*iterator == tab) {
690 tabs.erase(iterator);
691
692 delete tab;
693 break;
694 }
695 }
696
697 // Remove scene from existing scenes
698 for (auto iterator = scenes.begin(); iterator < scenes.end(); iterator++) {
699 if (*iterator == scene) {
700 scenes.erase(iterator);
701
702 delete scene;
703 return;
704 }
705 }

256void SequenceDiagramScene::sendMessage()
257{
258 // Create new messageline in scene
259 ObjectGraphicsItem *objectReciever = dynamic_cast<ObjectGraphicsItem *>(reciever);
260 MethodReference ref{""};
261 if(objectReciever)
262 {
263 std::vector<ClassMethod> methods = objectReciever->getObject()->getInstanceClass()->getMethods();
264 if(methods.size() > 0)
265 ref = MethodReference{&(objectReciever->getObject()->getInstanceClass()->getMethods()[0])};
266 else
267 ref = MethodReference{"UNKNOWN METHOD"};
268 }
269 else
270 ref = MethodReference{"UNKNOWN METHOD"};
271
272 //check if is possible to create new line
273 QString errorMsg = "";
274 if(!createMessagePossible(&errorMsg, reciever, sender, newMessageLineType))
275 {
276 QMessageBox msgBox;
277 msgBox.setText(errorMsg);
278 msgBox.setWindowTitle("Creation was not succesful");
279 msgBox.setIcon(QMessageBox::Critical);
280 msgBox.exec();
281 return;
282 }
283
284 Message *newMessage = new Message(ref, newMessageLineType, sender->getMessageNode(), reciever->getMessageNode(),0.0);
285
286 ClassReference classRef{""};
287 ObjectGraphicsItem *recieverObject = dynamic_cast<ObjectGraphicsItem *>(reciever);
288 if(recieverObject)
289 classRef = recieverObject->getObject()->getInstanceClass();
290
291 newMessageLine->initialize(sender, reciever, newMessage, classRef);
292 addItem(newMessageLine);
293
294 // Store new message in squence diagram
295 sequenceDiagram->addMessage(newMessage);
296 sender = nullptr;
297 reciever = nullptr;
298 newMessageLine = nullptr;
299
300 sceneUpdateObservable->sceneChanged();
301}

329QVector<ActivationGraphicsObjectBase *> SequenceDiagramScene::activationObjectsInScene()
330{
331 QList<QGraphicsItem*> sceneItems = items(Qt::AscendingOrder);
332 QVector<ActivationGraphicsObjectBase*> activationItems;
333 for(QGraphicsItem* item : sceneItems)
334 {
335 ActivationGraphicsObjectBase *activationObject = dynamic_cast<ActivationGraphicsObjectBase *>(item);
336 if(activationObject)
337 activationItems.push_back(activationObject);
338 }
339
340 //sort items by order
341 std::sort(activationItems.begin(), activationItems.end(),
342 [](ActivationGraphicsObjectBase * first, ActivationGraphicsObjectBase *second)

344 return first->order < second->order;
345 });

384bool SequenceDiagramScene::createMessagePossible(QString *errorMsg,
385 ActivationGraphicsObjectBase *reciever, ActivationGraphicsObjectBase *sender ,MessageType messageType)
386{
387 if(reciever == sender)
388 {
389 *errorMsg = "Reciever cannot be sender in one message!";
390 return false;
391 }
392
393 ActorGraphicsItem *actor = dynamic_cast<ActorGraphicsItem*>(reciever);
394 if(actor && (messageType == (MessageType)MessageType::CREATE || messageType == (MessageType)MessageType::DESTROY))
395 {
396 *errorMsg = "Actor cannot recieve any message of type <<create>> nor <<destroy>>!";
397 return false;
398 }
399 else
400 {
401 ObjectGraphicsItem *obj = dynamic_cast<ObjectGraphicsItem *>(reciever);
402 if(messageType == (MessageType)MessageType::DESTROY && obj->getDestroyMessage())
403 {
404 *errorMsg = "Object cannot recieve more than one destroy message!";
405 return false;
406 }
407 else if (messageType == (MessageType)MessageType::CREATE && obj->getCreateMessage())
408 {
409 *errorMsg = "Object cannot recieve more than one create message!";
410 return false;
411 }
412 }
413 return true;
414}

252void SequenceDiagramScene::sendMessage()
253{
254 // Create new messageline in scene
255 ObjectGraphicsItem *objectReciever = dynamic_cast<ObjectGraphicsItem *>(reciever);
256 MethodReference ref{""};
257 if(objectReciever)
258 {
259 std::vector<ClassMethod> methods = objectReciever->getObject()->getInstanceClass()->getMethods();
260 if(methods.size() > 0)
261 ref = MethodReference{&(objectReciever->getObject()->getInstanceClass()->getMethods()[0])};
262 else
263 ref = MethodReference{"UNKNOWN METHOD"};
264 }
265 else
266 ref = MethodReference{"UNKNOWN METHOD"};
267
268 //check if is possible to create new line
269 QString errorMsg = "";
270 if(!createMessagePossible(&errorMsg, reciever, sender, newMessageLineType))
271 {
272 QMessageBox msgBox;
273 msgBox.setText(errorMsg);
274 msgBox.setWindowTitle("Creation was not succesful");
275 msgBox.setIcon(QMessageBox::Critical);
276 msgBox.exec();
277 return;
278 }
279
280 Message *newMessage = new Message(ref, newMessageLineType, sender->getMessageNode(), reciever->getMessageNode(),0.0);
281
282 ClassReference classRef{""};
283 ObjectGraphicsItem *recieverObject = dynamic_cast<ObjectGraphicsItem *>(reciever);
284 if(recieverObject)
285 classRef = recieverObject->getObject()->getInstanceClass();
286
287 newMessageLine->initialize(sender, reciever, newMessage, classRef);
288 addItem(newMessageLine);
289
290 // Store new message in squence diagram
291 sequenceDiagram->addMessage(newMessage);
292 sender = nullptr;
293 reciever = nullptr;
294 newMessageLine = nullptr;
295
296 sceneUpdateObservable->sceneChanged();
297}

375bool SequenceDiagramScene::createMessagePossible(QString *errorMsg,
376 ActivationGraphicsObjectBase *reciever, ActivationGraphicsObjectBase *sender ,MessageType messageType)
377{
378 if(reciever == sender)
379 {
380 *errorMsg = "Reciever cannot be sender in one message!";
381 return false;
382 }
383
384 ActorGraphicsItem *actor = dynamic_cast<ActorGraphicsItem*>(reciever);
385 if(actor && (messageType == (MessageType)MessageType::CREATE || messageType == (MessageType)MessageType::DESTROY))
386 {
387 *errorMsg = "Actor cannot has create or destroy message!";
388 return false;
389 }
390 else
391 {
392 ObjectGraphicsItem *obj = dynamic_cast<ObjectGraphicsItem *>(reciever);
393 if(messageType == (MessageType)MessageType::DESTROY && obj->getDestroyMessage())
394 {
395 *errorMsg = "Object cannot has more than one destroy message!";
396 return false;
397 }
398 else if (messageType == (MessageType)MessageType::CREATE && obj->getCreateMessage())
399 {
400 *errorMsg = "Object cannot has more than one create message!";
401 return false;
402 }
403 }
404 return true;
405}

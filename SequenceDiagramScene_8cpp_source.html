238bool SequenceDiagramScene::createMessagePossible(QString *errorMsg, ActivationGraphicsObjectBase *receiver,
239 ActivationGraphicsObjectBase *sender , MessageType messageType)
240{
241 if (receiver == sender) {
242 *errorMsg = "Receiver cannot be newSender in one message!";
243
244 return false;
245 }
246
247 auto *actor = dynamic_cast<ActorGraphicsItem*>(receiver);
248 if (actor && (messageType == (MessageType)MessageType::CREATE || messageType == (MessageType)MessageType::DESTROY)) {
249 *errorMsg = "Actor cannot recieve any message of type <<create>> nor <<destroy>>!";
250 return false;
251 } else {
252 auto *obj = dynamic_cast<ObjectGraphicsItem *>(receiver);
253 if(messageType == (MessageType)MessageType::DESTROY && obj->getDestroyMessage()) {
254 *errorMsg = "Object cannot recieve more than one destroy message!";
255 return false;
256 } else if (messageType == (MessageType)MessageType::CREATE && obj->getCreateMessage()) {
257 *errorMsg = "Object cannot recieve more than one create message!";
258 return false;
259 }
260 }
261
262 return true;
263}

331void SequenceDiagramScene::sendMessage()
332{
333 // Create new message line in scene
334 auto *objectReceiver = dynamic_cast<ObjectGraphicsItem *>(newReceiver);
335 MethodReference ref{"UNKNOWN METHOD"};
336 if (objectReceiver) {
337 std::vector<ClassMethod> methods = objectReceiver->getObject()->getInstanceClass()->getMethods();
338 if (!methods.empty()) {
339 ref = MethodReference{&(objectReceiver->getObject()->getInstanceClass()->getMethods()[0])};
340 }
341 }
342
343 //check if is possible to create new line
344 QString errorMsg = "";
345 if (!createMessagePossible(&errorMsg, newReceiver, newSender, newMessageLineType)) {
346 QMessageBox msgBox;
347 msgBox.setText(errorMsg);
348 msgBox.setWindowTitle("Creation was not successful");
349 msgBox.setIcon(QMessageBox::Critical);
350 msgBox.exec();
351 return;
352 }
353
354 auto *newMessage = new Message(
355 ref,
356 newMessageLineType,
357 newSender->getMessageNode(),
358 newReceiver->getMessageNode(),
359 0.0
360 );
361
362 auto *receiverObject = dynamic_cast<ObjectGraphicsItem *>(newReceiver);
363 ClassReference classRef = receiverObject->getObject()->getInstanceClass();
364
365 newMessageLine->initialize(newSender, newReceiver, newMessage, classRef);
366 addItem(newMessageLine);
367
368 // Store new message in sequence diagram
369 sequenceDiagram.addMessage(newMessage);
370 newSender = nullptr;
371 newReceiver = nullptr;
372 newMessageLine = nullptr;
373
374 sceneUpdateObservable->sceneChanged();
375}

394QVector<ActivationGraphicsObjectBase *> SequenceDiagramScene::activationObjectsInScene()
395{
396 QList<QGraphicsItem*> sceneItems = items(Qt::AscendingOrder);
397 QVector<ActivationGraphicsObjectBase*> activationItems;
398 for (QGraphicsItem* item : sceneItems) {
399 auto *activationObject = dynamic_cast<ActivationGraphicsObjectBase *>(item);
400 if (activationObject) {
401 activationItems.push_back(activationObject);
402 }
403 }
404
405 //sort items by order
406 std::sort(activationItems.begin(), activationItems.end(),
407 [](ActivationGraphicsObjectBase * first, ActivationGraphicsObjectBase *second)
408 {
409 return first->order < second->order;
410 });
411 return activationItems;
412}

464void SequenceDiagramScene::redrawSequenceDiagram()
465{
466 std::unordered_map<Actor *, ActorGraphicsItem *> loadedActors;
467 std::unordered_map<Object *, ObjectGraphicsItem *> loadedObjects;
468
469 // Load actors
470 for (const auto objActor: sequenceDiagram.getActors()) {
471 auto *guiActor = new ActorGraphicsItem(sceneUpdateObservable, objActor);
472
473 connect(&(guiActor->emitter), &ActivationObjectEmitter::objectPressed, this, &SequenceDiagramScene::objectPressed);
474 connect(&(guiActor->emitter), &ActivationObjectEmitter::removeObject, this, &SequenceDiagramScene::removeObject);
475 connect(&(guiActor->emitter), &ActivationObjectEmitter::moveLeft, this, &SequenceDiagramScene::moveObjectLeft);
476 connect(&(guiActor->emitter), &ActivationObjectEmitter::moveRight, this, &SequenceDiagramScene::moveObjectRight);
477
478 addItem(guiActor);
479 loadedActors.insert({objActor, guiActor});
480 }
481
482 // Load objects
483 for (const auto objObject: sequenceDiagram.getObjects()) {
484 auto *guiObject = new ObjectGraphicsItem(sceneUpdateObservable, objObject, classDiagram);
485
486 connect(&(guiObject->emitter), &ActivationObjectEmitter::objectPressed, this, &SequenceDiagramScene::objectPressed);
487 connect(&(guiObject->emitter), &ActivationObjectEmitter::removeObject, this, &SequenceDiagramScene::removeObject);
488 connect(&(guiObject->emitter), &ActivationObjectEmitter::moveLeft, this, &SequenceDiagramScene::moveObjectLeft);
489 connect(&(guiObject->emitter), &ActivationObjectEmitter::moveRight, this, &SequenceDiagramScene::moveObjectRight);
490
491
492 addItem(guiObject);
493 loadedObjects.insert({objObject, guiObject});
494 }
495
496 sortActivationItems();
497
498 // Load messages
499 for (const auto objMessage: sequenceDiagram.getMessages()) {
500 // Message sender
501 MessageNode *messageSender = objMessage->getMessageSender();
502 ActivationGraphicsObjectBase *senderGuiItem;
503 if (typeid(*messageSender) == typeid(Actor)) {
504 auto actor = dynamic_cast<Actor *>(messageSender);
505 senderGuiItem = dynamic_cast<ActivationGraphicsObjectBase *>(loadedActors.find(actor)->second);
506 } else if (typeid(*messageSender) == typeid(Object)) {
507 auto object = dynamic_cast<Object *>(messageSender);
508 senderGuiItem = dynamic_cast<ActivationGraphicsObjectBase *>(loadedObjects.find(object)->second);
509 } else {
510 throw std::logic_error{"Unknown message sender type"};
511 }
512
513 // Message recipient
514 MessageNode *messageRecipient = objMessage->getMessageRecipient();
515 ActivationGraphicsObjectBase *recipientGuiItem;
516 ClassReference calleeClassReference{"ACTOR"};
517 if (typeid(*messageRecipient) == typeid(Actor)) {
518 auto actor = dynamic_cast<Actor *>(messageRecipient);
519 recipientGuiItem = dynamic_cast<ActivationGraphicsObjectBase *>(loadedActors.find(actor)->second);
520 } else if (typeid(*messageRecipient) == typeid(Object)) {
521 auto object = dynamic_cast<Object *>(messageRecipient);
522 recipientGuiItem = dynamic_cast<ActivationGraphicsObjectBase *>(loadedObjects.find(object)->second);
523
524 calleeClassReference = object->getInstanceClass();
525 } else {
526 throw std::logic_error{"Unknown message recipient type"};
527 }
528
529 // Create and allocate message
530 switch (objMessage->getType()) {
531 case MessageType::SYNC:
532 newMessageLine = new SyncMessageLine{sceneUpdateObservable, &sequenceDiagram};
533 break;
534 case MessageType::ASYNC:
535 newMessageLine = new AsyncMessageLine{sceneUpdateObservable, &sequenceDiagram};
536 break;
537 case MessageType::CREATE:
538 newMessageLine = new CreateMessageLine{sceneUpdateObservable, &sequenceDiagram};
539 break;
540 case MessageType::DESTROY:
541 newMessageLine = new DestroyMessageLine{sceneUpdateObservable, &sequenceDiagram};
542 break;
543 case MessageType::REPLY:
544 newMessageLine = new ReplyMessageLine{sceneUpdateObservable, &sequenceDiagram};
545 break;
546 }
547
548 newMessageLine->initialize(senderGuiItem, recipientGuiItem, objMessage, calleeClassReference);
549 addItem(newMessageLine);
550 }
551}

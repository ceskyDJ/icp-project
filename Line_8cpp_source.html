85QPointF Line::getIntersectPoint(QLineF connectingLine, ClassNode *node) const
86{
87 QRectF rect = node->boundingRect();
88 qreal xPos = node->x() + rect.x();
89 qreal yPos = node->y() + rect.y();
90 std::vector<QPointF> connectPoints{QPointF{xPos, yPos},
91 QPointF{ xPos + rect.width(), yPos},
92 QPointF{ xPos + rect.width(), yPos + rect.height()},
93 QPointF{ xPos, yPos + rect.height()},
94 QPointF{ xPos, yPos}
95 };
96 QPointF toReturn = QPointF{-1000, -1000};
97 for(int i = 0; (size_t)i < connectPoints.size() - 1; ++i)
98 {
99 if(connectingLine.intersects(QLineF{connectPoints[i], connectPoints[i + 1]}, &toReturn) == QLineF::BoundedIntersection)
100 return toReturn;
101 }
102 return toReturn;
103}

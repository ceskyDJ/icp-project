94 qreal yPos = node->y() + rect.y();

308void Line::movePointsByConnectionCount(QPointF *firstPoint, QPointF *secondPoint,
309 ClassNode *firstNode, ClassNode *secondNode) const
310{
311 int lineIndex;
312 int connectionCount = firstNode->getNumberOfConnectionsWithNode(secondNode, this, &lineIndex);
313 if(connectionCount == 1)//optimalization - no need to count all that things if there is just one connection
314 return;
315
316 QLineF line{*firstPoint, *secondPoint};
317 QRectF firstRect = firstNode->boundingRect();
318 firstRect.translate(firstNode->pos());
319
320 QRectF secondRect = secondNode->boundingRect();
321 secondRect.translate(secondNode->pos());
322
323 QLineF perpen = line.normalVector();
324 perpen.setLength(firstRect.width() + firstRect.height()); // nessacary for situation when line is shorter than node
325 QLineF firstIntLine = qrectIntersectsLine(firstRect, perpen);
326 QLineF reversedLine{line.p2(), line.p1()};
327 perpen = reversedLine.normalVector();
328 perpen.setLength(secondRect.width() + secondRect.height());
329 QLineF secondIntLine = qrectIntersectsLine(secondRect, perpen);
330
331 setEndPoints(&firstIntLine);
332 movePointInLine(firstIntLine, firstPoint, connectionCount, lineIndex);
333 QLineF reversedSecondIntLine{secondIntLine.p2(), secondIntLine.p1()};
334 setEndPoints(&reversedSecondIntLine);
335 movePointInLine(reversedSecondIntLine, secondPoint, connectionCount, lineIndex);
336}

367QLineF Line::qrectIntersectsLine(QRectF rect, QLineF line) const
368{
369 QVector<QPointF> rectPoints{ rect.topLeft(),
370 QPointF{rect.x() + rect.width(), rect.y()},
371 QPointF{rect.x() + rect.width(), rect.y() + rect.height()},
372 QPointF{rect.x(), rect.y() + rect.height()},
373 rect.topLeft()
374 };
375
376 QPointF intPointOne;
377 QPointF intPointTwo;
378 for(int i = 0; i < rectPoints.size() - 1; i++)
379 {
380 if(line.intersects(QLineF{rectPoints[i], rectPoints[i + 1]}, &intPointOne) == QLineF::BoundedIntersection)
381 {
382 if(i < 2)
383 line.intersects(QLineF(rectPoints[i + 2], rectPoints[i + 3]), &intPointTwo);
384 else
385 line.intersects(QLineF(rectPoints[i - 2], rectPoints[i - 1]), &intPointTwo);
386 break;
387 }
388 }
389
390 return QLineF{intPointOne, intPointTwo};
391}

136inline std::vector<Relationship *> deepCloneRelationships(
137 const std::vector<Relationship *> &sourceRelationships,
138 const std::map<Class *, Class *> &classMap
139) {
140 std::vector<Relationship *> clonedRelationships{};
141
142 // Deep clone relationships
143 for (const auto item: sourceRelationships) {
144 Relationship *relationshipBackup;
145
146 // It must be done separately, because different relationships have different attributes
147 const std::type_info &relationshipType = typeid(*item);
148 if (relationshipType == typeid(Aggregation)) {
149 auto aggregation = dynamic_cast<Aggregation *>(item);
150 relationshipBackup = new Aggregation{*aggregation};
151 } else if (relationshipType == typeid(Composition)) {
152 auto composition = dynamic_cast<Composition *>(item);
153 relationshipBackup = new Composition{*composition};
154 } else if (relationshipType == typeid(DirectedAssociation)) {
155 auto directedAssociation = dynamic_cast<DirectedAssociation *>(item);
156 relationshipBackup = new DirectedAssociation{*directedAssociation};
157 } else if (relationshipType == typeid(Generalization)) {
158 auto generalization = dynamic_cast<Generalization *>(item);
159 relationshipBackup = new Generalization{*generalization};
160 } else if (relationshipType == typeid(Realization)) {
161 auto realization = dynamic_cast<Realization *>(item);
162 relationshipBackup = new Realization{*realization};
163 } else if (relationshipType == typeid(UndirectedAssociation)) {
164 auto undirectedAssociation = dynamic_cast<UndirectedAssociation *>(item);
165 relationshipBackup = new UndirectedAssociation{*undirectedAssociation};
166 } else {
167 throw std::logic_error{"Unknown relationship"};
168 }
169
170 Class *newFirstClass = classMap.find(relationshipBackup->getFirstClass())->second;
171 Class *newSecondClass = classMap.find(relationshipBackup->getSecondClass())->second;
172
173 relationshipBackup->setFirstClass(newFirstClass);
174 relationshipBackup->setSecondClass(newSecondClass);
175
176 clonedRelationships.push_back(relationshipBackup);
177 }
178
179 return clonedRelationships;

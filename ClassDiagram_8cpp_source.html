172inline std::vector<Relationship *> deepCloneRelationships(
173 const std::vector<Relationship *> &sourceRelationships,
174 const std::map<Class *, Class *> &classMap
175) {
176 std::vector<Relationship *> clonedRelationships{};
177
178 // Deep clone relationships
179 for (const auto item: sourceRelationships) {
180 Relationship *relationshipBackup;
181
182 // It must be done separately, because different relationships have different attributes
183 const std::type_info &relationshipType = typeid(*item);
184 if (relationshipType == typeid(Aggregation)) {
185 auto aggregation = dynamic_cast<Aggregation *>(item);
186 relationshipBackup = new Aggregation{*aggregation};
187 } else if (relationshipType == typeid(Composition)) {
188 auto composition = dynamic_cast<Composition *>(item);
189 relationshipBackup = new Composition{*composition};
190 } else if (relationshipType == typeid(DirectedAssociation)) {
191 auto directedAssociation = dynamic_cast<DirectedAssociation *>(item);
192 relationshipBackup = new DirectedAssociation{*directedAssociation};
193 } else if (relationshipType == typeid(Generalization)) {
194 auto generalization = dynamic_cast<Generalization *>(item);
195 relationshipBackup = new Generalization{*generalization};
196 } else if (relationshipType == typeid(Realization)) {
197 auto realization = dynamic_cast<Realization *>(item);
198 relationshipBackup = new Realization{*realization};
199 } else if (relationshipType == typeid(UndirectedAssociation)) {
200 auto undirectedAssociation = dynamic_cast<UndirectedAssociation *>(item);
201 relationshipBackup = new UndirectedAssociation{*undirectedAssociation};
202 } else {
203 throw std::logic_error{"Unknown relationship"};
204 }
205
206 Class *newFirstClass = classMap.find(relationshipBackup->getFirstClass())->second;
207 Class *newSecondClass = classMap.find(relationshipBackup->getSecondClass())->second;
208
209 relationshipBackup->setFirstClass(newFirstClass);
210 relationshipBackup->setSecondClass(newSecondClass);
211
212 clonedRelationships.push_back(relationshipBackup);
213 }
214
215 return clonedRelationships;
